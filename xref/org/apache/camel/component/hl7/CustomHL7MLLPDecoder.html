17  package org.apache.camel.component.hl7;
18  
19  import org.apache.mina.core.buffer.IoBuffer;
20  import org.apache.mina.core.session.IoSession;
21  import org.apache.mina.filter.codec.CumulativeProtocolDecoder;
22  import org.apache.mina.filter.codec.ProtocolDecoderOutput;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import java.nio.charset.CharacterCodingException;
27  import java.nio.charset.CharsetDecoder;
28  
29  /**
30   * HL7MLLPDecoder that is aware that a HL7 message can span several buffers.
31   * In addition, it avoids rescanning packets by keeping state in the IOSession.
32   *
33   * This decoder addresses a decoding error that might occur when only the end bytes
34   * are transferred in a different frame.
35   */
36  class CustomHL7MLLPDecoder extends CumulativeProtocolDecoder {
37  
38      private static final Logger LOG = LoggerFactory.getLogger(CustomHL7MLLPDecoder.class);
39      private static final String DECODER_STATE = CustomHL7MLLPDecoder.class.getName() + ".STATE";
40      private static final String CHARSET_DECODER = CustomHL7MLLPDecoder.class.getName() + ".charsetdecoder";
41  
42      private CustomHL7MLLPConfig config;
43  
44      CustomHL7MLLPDecoder(CustomHL7MLLPConfig config) {
45          this.config = config;
46      }
47  
48  
49      @Override
50      protected boolean doDecode(IoSession session, IoBuffer in, ProtocolDecoderOutput out) throws Exception {
51  
52          // Get the state of the current message and
53          // Skip what we have already scanned before
54          DecoderState state = decoderState(session);
55          in.position(state.current());
56  
57          LOG.debug("Received data, checking from position {} to {}", in.position(), in.limit());
58  
59          while (in.hasRemaining()) {
60              int previousPosition = in.position();
61              byte current = in.get();
62  
63              if (current == config.getStartByte()) {
64                  state.markStart(previousPosition);
65              }
66  
67              if (state.previous() == config.getEndByte1() && current == config.getEndByte2()) {
68                  if (state.isStarted()) {
69                      int currentPosition = in.position();
70                      int currentLimit = in.limit();
71                      LOG.debug("Message ends at position {} with length {}", currentPosition, currentPosition - state.start());
72                      in.position(state.start());
73                      in.limit(currentPosition);
74                      LOG.debug("Set start to position {} and limit to {}", in.position(), in.limit());
75                      try {
76                          out.write(config.isProduceString()
77                                  ? parseMessageToString(in, charsetDecoder(session))
78                                  : parseMessageToByteArray(in));
79                          return true;
80                      } finally {
81                          in.position(currentPosition);
82                          in.limit(currentLimit);
83                          state.reset();
84                      }
85                  } else {
86                      LOG.warn("Ignoring message end at position {} until start byte has been seen.", previousPosition);
87                  }
88  
89              }
90              // Remember previous byte in state object because the buffer could
91              // be theoretically exhausted right between the two end bytes
92              state.markPrevious(current);
93          }
94  
95          // Could not find a complete message in the buffer.
96          // Reset to the initial position (just as nothing had been read yet)
97          // and return false so that this method is called again with more data.
98          LOG.debug("No complete message yet at position {} ", in.position());
99          state.markCurrent(in.position());
100         in.position(0);
101         return false;
102     }
103 
104     // Make a defensive byte copy (the buffer will be reused)
105     // and omit the start and the two end bytes of the MLLP message
106     // returning a byte array
107     private Object parseMessageToByteArray(IoBuffer buf) throws CharacterCodingException {
108         int len = buf.limit() - 3;
109         LOG.debug("Making {} bytes", len);
110         byte[] dst = new byte[len];
111         buf.skip(1); // skip start byte
112         buf.get(dst, 0, len);
113         buf.skip(2); // skip end bytes
114 
115         // Only do this if conversion is enabled
116         if (config.isConvertLFtoCR()) {
117             LOG.debug("Replacing LF by CR");
118             for (int i = 0; i < dst.length; i++) {
119                 if (dst[i] == (byte) '\n') {
120                     dst[i] = (byte) '\r';
121                 }
122             }
123         }
124         return dst;
125     }
126 
127     // Make a defensive byte copy (the buffer will be reused)
128     // and omit the start and the two end bytes of the MLLP message
129     // returning a String
130     private Object parseMessageToString(IoBuffer buf, CharsetDecoder decoder) throws CharacterCodingException {
131         int len = buf.limit() - 3;
132         LOG.debug("Making string of length {} using charset {}", len, decoder.charset());
133         buf.skip(1); // skip start byte
134         String message = buf.getString(buf.limit() - 3, decoder);
135         buf.skip(2); // skip end bytes
136 
137         // Only do this if conversion is enabled
138         if (config.isConvertLFtoCR()) {
139             LOG.debug("Replacing LF by CR");
140             message = message.replace('\n', '\r');
141         }
142         return message;
143     }
144 
145     @Override
146     public void dispose(IoSession session) throws Exception {
147         session.removeAttribute(DECODER_STATE);
148         session.removeAttribute(CHARSET_DECODER);
149     }
150 
151     private CharsetDecoder charsetDecoder(IoSession session) {
152         synchronized (session) {
153             CharsetDecoder decoder = (CharsetDecoder) session.getAttribute(CHARSET_DECODER);
154             if (decoder == null) {
155                 decoder = config.getCharset().newDecoder()
156                     .onMalformedInput(config.getMalformedInputErrorAction())
157                     .onUnmappableCharacter(config.getUnmappableCharacterErrorAction());
158                 session.setAttribute(CHARSET_DECODER, decoder);
159             }
160             return decoder;
161         }
162     }
163 
164     private DecoderState decoderState(IoSession session) {
165         synchronized (session) {
166             DecoderState decoderState = (DecoderState) session.getAttribute(DECODER_STATE);
167             if (decoderState == null) {
168                 decoderState = new DecoderState();
169                 session.setAttribute(DECODER_STATE, decoderState);
170             }
171             return decoderState;
172         }
173     }
174 
175     /**
176      * Holds the state of the decoding process
177      */
178     private static class DecoderState {
179         private int startPos = -1;
180         private int currentPos;
181         private byte previousByte;
182 
183         void reset() {
184             startPos = -1;
185             currentPos = 0;
186             previousByte = 0;
187         }
188 
189         void markStart(int position) {
190             if (isStarted()) {
191                 LOG.warn("Ignoring message start at position {} before previous message has ended.", position);
192             } else {
193                 startPos = position;
194                 LOG.debug("Message starts at position {}", startPos);
195             }
196         }
197 
198         void markCurrent(int position) {
199             currentPos = position;
200         }
201 
202         void markPrevious(byte previous) {
203             previousByte = previous;
204         }
205 
206         public int start() {
207             return startPos;
208         }
209 
210         public int current() {
211             return currentPos;
212         }
213 
214         public byte previous() {
215             return previousByte;
216         }
217 
218         public boolean isStarted() {
219             return startPos >= 0;
220         }
221     }
