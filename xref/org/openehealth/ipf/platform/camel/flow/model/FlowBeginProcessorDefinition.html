16  package org.openehealth.ipf.platform.camel.flow.model;
17  
18  import org.apache.camel.CamelContext;
19  import org.apache.camel.spi.Metadata;
20  import org.apache.camel.spi.RouteContext;
21  import org.openehealth.ipf.commons.flow.FlowManager;
22  import org.openehealth.ipf.commons.flow.transfer.FlowInfo;
23  import org.openehealth.ipf.platform.camel.core.util.Contexts;
24  import org.openehealth.ipf.platform.camel.flow.ReplayStrategyRegistry;
25  import org.openehealth.ipf.platform.camel.flow.process.FlowBeginProcessor;
26  import org.openehealth.ipf.platform.camel.flow.process.FlowProcessor;
27  
28  import javax.xml.bind.annotation.XmlAccessType;
29  import javax.xml.bind.annotation.XmlAccessorType;
30  import javax.xml.bind.annotation.XmlAttribute;
31  import javax.xml.bind.annotation.XmlRootElement;
32  
33  /**
34   * @author Martin Krasser
35   */
36  @Metadata(label = "ipf,eip,flow")
37  @XmlRootElement(name = "initFlow")
38  @XmlAccessorType(XmlAccessType.FIELD)
39  public class FlowBeginProcessorDefinition extends FlowProcessorDefinition {
40  
41      @XmlAttribute
42      private final String identifier;
43      @XmlAttribute
44      private String application;
45      @XmlAttribute
46      private String replayErrorUri;
47      @XmlAttribute
48      private int expectedAckCount = FlowInfo.ACK_COUNT_EXPECTED_UNDEFINED;
49      
50      public FlowBeginProcessorDefinition() {
51          this("flow");
52      }
53  
54      public FlowBeginProcessorDefinition(String identifier) {
55          this.identifier = identifier;
56      }
57  
58      /**
59       * Assigns an application name to created flows
60       * @param application
61       *          the name of the application
62       */
63      public FlowProcessorDefinition application(String application) {
64          this.application = application;
65          return this;
66      }
67      
68      /**
69       * Defines the endpoint URI for error messages in case a flow replay fails
70       * @param replayErrorUri
71       *          the endpoint URI
72       */
73      public FlowBeginProcessorDefinition replayErrorHandler(String replayErrorUri) {
74          this.replayErrorUri = replayErrorUri;
75          return this;
76      }
77      
78      /**
79       * Defines the number of acknowledgments that are expected for a created flow
80       * @param expectedAckCount
81       *          the expected number of acknowledgments 
82       */
83      public FlowBeginProcessorDefinition expectedAckCount(int expectedAckCount) {
84          this.expectedAckCount = expectedAckCount;
85          return this;
86      }
87      
88      @Override
89      public String toString() {
90          return "FlowBeginProcessor[" + identifier + " -> " + getOutputs() + "]";
91      }
92  
93      @Override
94      public String getShortName() {
95          return "flowBeginProcessor";
96      }
97  
98      @Override
99      public FlowProcessor doCreateProcessor(RouteContext routeContext) throws Exception {
100         FlowBeginProcessor processor = createFlowBeginProcessor(routeContext);
101         processor
102             .identifier(identifier)
103             .application(application)
104             .expectedAckCount(expectedAckCount)
105             .replayErrorHandler(replayErrorUri)
106             .register();
107         return processor;
108     }
109 
110     private static FlowBeginProcessor createFlowBeginProcessor(RouteContext routeContext) {
111         CamelContext camelContext = routeContext.getCamelContext();
112         
113         // Try to obtain a FlowBeginProcessor bean (its definition is optional)
114         FlowBeginProcessor processor = Contexts.beanOrNull(FlowBeginProcessor.class, camelContext);
115         
116         if (processor != null) {
117             return processor;
118         }
119         
120         // No FlowBeginProcessor bean found so let's create one. We need a
121         // - reference to a ReplayStrategyRegistry
122         // - reference to a FlowManager
123         processor = new FlowBeginProcessor();
124         processor.setCamelContext(camelContext);
125         processor.setFlowManager(Contexts.bean(FlowManager.class, camelContext));
126         processor.setRegistry(Contexts.bean(ReplayStrategyRegistry.class, camelContext));
127         return processor;
128     }
129     
130 }
