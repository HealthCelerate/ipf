40  /**
41   * Configuration of a FHIR endpoint instance
42   *
43   * @author Christian Ohr
44   * @since 3.1
45   */
46  @UriParams
47  public class FhirEndpointConfiguration<AuditDatasetType extends FhirAuditDataset> extends InterceptableEndpointConfiguration {
48  
49      static final String STRICT = "strict";
50      static final String LENIENT = "lenient";
51      static final String LAZY_LOAD_BUNDLES = "lazyLoadBundles";
52      static final String CACHE_BUNDLES = "cacheBundles";
53  
54      @Getter
55      private final String path;
56  
57      @Getter
58      private FhirContext context;
59  
60      // Consumer only
61  
62      @Getter
63      @UriParam(defaultValue = "false")
64      private boolean audit = false;
65  
66      @Getter
67      @UriParam(defaultValue = "FhirServlet")
68      private String servletName = IpfFhirServlet.DEFAULT_SERVLET_NAME;
69  
70      @Getter
71      @UriParam
72      private AbstractPlainProvider resourceProvider;
73  
74      // Producer only
75  
76      @UriParam
77      private ClientRequestFactory<? extends IClientExecutable<?, ?>> clientRequestFactory;
78  
79  
80      @Getter
81      @UriParam
82      private List<HapiClientInterceptorFactory> hapiClientInterceptorFactories;
83  
84      @Getter
85      @UriParam
86      private List<HapiServerInterceptorFactory> hapiServerInterceptorFactories;
87  
88      /**
89       * If this is true, all paging requests are routed into the route (see {@link org.openehealth.ipf.commons.ihe.fhir.LazyBundleProvider}
90       * fo details. Otherwise, all results are fetched once and cached in order to serve subsequent paging requests.
91       */
92      @Getter
93      @UriParam
94      private boolean lazyLoadBundles = false;
95  
96      @Getter
97      private FhirSecurityInformation securityInformation;
98  
99  
100     /**
101      * Only considered if {@link #lazyLoadBundles} is true. The (partial) results of paging requests are cached so that subsequent
102      * requests only fetch resources that have not yet been requested.
103      */
104     @Getter
105     @UriParam
106     private boolean cacheBundles = true;
107 
108     protected FhirEndpointConfiguration(FhirComponent<AuditDatasetType> component, String path, Map<String, Object> parameters) throws Exception {
109         super(component, parameters);
110         this.path = path;
111         this.context = component.createFhirContext();
112         audit = component.getAndRemoveParameter(parameters, "audit", boolean.class, true);
113         servletName = component.getAndRemoveParameter(parameters, "servletName", String.class, IpfFhirServlet.DEFAULT_SERVLET_NAME);
114         resourceProvider = component.getAndRemoveOrResolveReferenceParameter(
115                 parameters, "resourceProvider", AbstractPlainProvider.class, null);
116         clientRequestFactory = component.getAndRemoveOrResolveReferenceParameter(
117                 parameters, "clientRequestFactory", ClientRequestFactory.class, null);
118         hapiClientInterceptorFactories = component.getAndRemoveOrResolveReferenceParameter(
119                 parameters, "hapiClientInterceptorFactories", List.class);
120         // TODO: make use of use hapiServerInterceptorFactories
121         hapiServerInterceptorFactories = component.getAndRemoveOrResolveReferenceParameter(
122                 parameters, "hapiServerInterceptorFactories", List.class);
123 
124 
125         String parserErrorHandling = component.getAndRemoveParameter(parameters, "validation", String.class, "lenient");
126         if (STRICT.equals(parserErrorHandling)) {
127             context.setParserErrorHandler(new StrictErrorHandler());
128         } else if (!LENIENT.equals(parserErrorHandling)) {
129             throw new IllegalArgumentException("Validation must be either " + LENIENT + " (default) or " + STRICT);
130         }
131         Integer connectTimeout = component.getAndRemoveParameter(parameters, "connectionTimeout", Integer.class);
132         if (connectTimeout != null) {
133             setConnectTimeout(connectTimeout);
134         }
135         Integer timeout = component.getAndRemoveParameter(parameters, "timeout", Integer.class);
136         if (timeout != null) {
137             setTimeout(timeout);
138         }
139 
140         setDisableServerValidation(component.getAndRemoveParameter(parameters, "disableServerValidation", Boolean.class, false));
141         lazyLoadBundles = component.getAndRemoveParameter(parameters, LAZY_LOAD_BUNDLES, Boolean.class, false);
142         cacheBundles = component.getAndRemoveParameter(parameters, CACHE_BUNDLES, Boolean.class, true);
143 
144         // Security stuff
145         SSLContextParameters sslContextParameters = component.getAndRemoveOrResolveReferenceParameter(
146                 parameters, "sslContextParameters", SSLContextParameters.class);
147         HostnameVerifier hostnameVerifier = component.getAndRemoveOrResolveReferenceParameter(
148                 parameters, "hostnameVerifier", HostnameVerifier.class);
149         boolean secure = component.getAndRemoveParameter(parameters, "secure", Boolean.class, false);
150         String username = component.getAndRemoveParameter(parameters, "username", String.class);
151         String password = component.getAndRemoveParameter(parameters, "password", String.class);
152 
153         if (sslContextParameters == null) {
154             Map<String, SSLContextParameters> sslContextParameterMap = component.getCamelContext().getRegistry().findByTypeWithName(SSLContextParameters.class);
155             if (sslContextParameterMap.size() == 1) {
156                 Map.Entry<String, SSLContextParameters> entry = sslContextParameterMap.entrySet().iterator().next();
157                 sslContextParameters = entry.getValue();
158             } else if (sslContextParameterMap.size() > 1) {
159                 throw new AmbiguousBeanException(SSLContextParameters.class);
160             }
161         }
162         this.securityInformation = new FhirSecurityInformation(
163                 secure,
164                 sslContextParameters != null ?
165                         sslContextParameters.createSSLContext(component.getCamelContext()) :
166                         null,
167                 hostnameVerifier,
168                 username,
169                 password);
170     }
171 
172     public <T extends IClientExecutable<?, ?>> ClientRequestFactory<T> getClientRequestFactory() {
173         return (ClientRequestFactory<T>) clientRequestFactory;
174     }
175 
176     public void setConnectTimeout(int connectTimeout) {
177         context.getRestfulClientFactory().setConnectTimeout(connectTimeout);
178     }
179 
180     public void setTimeout(int timeout) {
181         context.getRestfulClientFactory().setSocketTimeout(timeout);
182     }
183 
184     /**
185      * By default, the client will query the server before the very first operation to download the server's conformance/metadata
186      * statement and verify that the server is appropriate for the given client.
187      * This check is only done once per server endpoint for a given FhirContext and is useful to prevent bugs or unexpected behaviour
188      * when talking to servers.
189      * <p>
190      * It may introduce unneccesary overhead however in circumstances where the client and server are known to be compatible.
191      * Setting this to true disables this check.
192      */
193     public void setDisableServerValidation(boolean disableServerValidation) {
194         context.getRestfulClientFactory().setServerValidationMode(disableServerValidation ?
195                 ServerValidationModeEnum.NEVER :
196                 ServerValidationModeEnum.ONCE);
197     }
198 }
