17  package org.openehealth.ipf.platform.camel.ihe.mllp.core;
18  
19  import lombok.Getter;
20  
21  import java.util.Map;
22  import org.openehealth.ipf.commons.ihe.hl7v2.storage.UnsolicitedFragmentationStorage;
23  import org.openehealth.ipf.commons.ihe.hl7v2.storage.InteractiveContinuationStorage;
24  
25  /**
26   * Camel endpoint configuration for MLLP-based eHealth transactions (like IHE PIX, PDQ, XAD-PID, etc.).
27   * @author Dmytro Rud
28   */
29  public class MllpTransactionEndpointConfiguration extends MllpEndpointConfiguration {
30      private static final long serialVersionUID = -6154765290339153487L;
31  
32      @Getter private final boolean supportUnsolicitedFragmentation;
33      @Getter private final int unsolicitedFragmentationThreshold;
34      @Getter private final UnsolicitedFragmentationStorage unsolicitedFragmentationStorage;
35  
36      @Getter private final boolean supportInteractiveContinuation;
37      @Getter private final int interactiveContinuationDefaultThreshold;
38      @Getter private final InteractiveContinuationStorage interactiveContinuationStorage;
39      @Getter private final boolean autoCancel;
40  
41      /**
42       * @deprecated
43       */
44      protected MllpTransactionEndpointConfiguration(MllpComponent<MllpTransactionEndpointConfiguration> component, Map<String, Object> parameters) throws Exception {
45          this(component, UNKNOWN_URI, parameters);
46      }
47  
48      protected MllpTransactionEndpointConfiguration(MllpComponent<MllpTransactionEndpointConfiguration> component, String uri, Map<String, Object> parameters) throws Exception {
49          super(component, uri, parameters);
50  
51          supportUnsolicitedFragmentation = component.getAndRemoveParameter(
52                  parameters, "supportUnsolicitedFragmentation", boolean.class, false);
53          unsolicitedFragmentationThreshold = component.getAndRemoveParameter(
54                  parameters, "unsolicitedFragmentationThreshold", int.class, -1);            // >= 3 segments
55  
56          unsolicitedFragmentationStorage = component.resolveAndRemoveReferenceParameter(
57                  parameters,
58                  "unsolicitedFragmentationStorage",
59                  UnsolicitedFragmentationStorage.class);
60  
61          supportInteractiveContinuation = component.getAndRemoveParameter(
62                  parameters, "supportInteractiveContinuation", boolean.class, false);
63          interactiveContinuationDefaultThreshold = component.getAndRemoveParameter(
64                  parameters, "interactiveContinuationDefaultThreshold", int.class, -1);      // >= 1 data record
65  
66          interactiveContinuationStorage = component.resolveAndRemoveReferenceParameter(
67                          parameters,
68                          "interactiveContinuationStorage",
69                          InteractiveContinuationStorage.class);
70  
71          autoCancel = component.getAndRemoveParameter(parameters, "autoCancel", boolean.class, false);
72      }
73  
74  }
