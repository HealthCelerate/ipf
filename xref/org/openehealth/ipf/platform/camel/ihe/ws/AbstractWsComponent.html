16  package org.openehealth.ipf.platform.camel.ihe.ws;
17  
18  import org.apache.camel.impl.DefaultComponent;
19  import org.apache.cxf.feature.AbstractFeature;
20  import org.apache.cxf.interceptor.AbstractBasicInterceptorProvider;
21  import org.apache.cxf.interceptor.Interceptor;
22  import org.apache.cxf.interceptor.InterceptorProvider;
23  import org.apache.cxf.message.Message;
24  import org.openehealth.ipf.commons.ihe.core.atna.AuditStrategy;
25  import org.openehealth.ipf.commons.ihe.ws.WsInteractionId;
26  import org.openehealth.ipf.commons.ihe.ws.WsTransactionConfiguration;
27  import org.openehealth.ipf.commons.ihe.ws.cxf.audit.WsAuditDataset;
28  import org.openehealth.ipf.platform.camel.ihe.atna.AuditableComponent;
29  
30  import java.util.List;
31  import java.util.Map;
32  
33  /**
34   * Base component class for Web Service-based IHE components.
35   *
36   * @param <AuditDatasetType> audit type
37   * @param <ConfigType>       configuration type
38   * @author Dmytro Rud
39   */
40  abstract public class AbstractWsComponent<AuditDatasetType extends WsAuditDataset, ConfigType extends WsTransactionConfiguration, InteractionIdType extends WsInteractionId<ConfigType>>
41          extends DefaultComponent implements AuditableComponent<AuditDatasetType> {
42  
43      private final InteractionIdType interactionId;
44  
45      public AbstractWsComponent(InteractionIdType interactionId) {
46          this.interactionId = interactionId;
47      }
48  
49      public InteractionIdType getInteractionId() {
50          return interactionId;
51      }
52  
53      protected InterceptorProvider getCustomInterceptors(Map<String, Object> parameters) {
54          AbstractBasicInterceptorProvider provider = new AbstractBasicInterceptorProvider() {
55          };
56          provider.setInInterceptors(castList(resolveAndRemoveReferenceListParameter(
57                  parameters, "inInterceptors", Interceptor.class)));
58          provider.setInFaultInterceptors(castList(resolveAndRemoveReferenceListParameter(
59                  parameters, "inFaultInterceptors", Interceptor.class)));
60          provider.setOutInterceptors(castList(resolveAndRemoveReferenceListParameter(
61                  parameters, "outInterceptors", Interceptor.class)));
62          provider.setOutFaultInterceptors(castList(resolveAndRemoveReferenceListParameter(
63                  parameters, "outFaultInterceptors", Interceptor.class)));
64  
65          return provider;
66      }
67  
68      @SuppressWarnings({"unchecked", "rawtypes"})
69      private List<Interceptor<? extends Message>> castList(
70              List<Interceptor> param) {
71          return (List<Interceptor<? extends Message>>) (List<?>) param;
72      }
73  
74      protected List<AbstractFeature> getFeatures(Map<String, Object> parameters) {
75          return resolveAndRemoveReferenceListParameter(parameters, "features", AbstractFeature.class);
76      }
77  
78      protected List<String> getSchemaLocations(Map<String, Object> parameters) {
79          return resolveAndRemoveReferenceListParameter(parameters, "schemaLocations", String.class);
80      }
81  
82      protected Map<String, Object> getProperties(Map<String, Object> parameters) {
83          List<Map> mapList = resolveAndRemoveReferenceListParameter(parameters, "properties", Map.class);
84          return (mapList != null && mapList.size() == 1) ? mapList.get(0) : null;
85      }
86  
87      @Override
88      public AuditStrategy<AuditDatasetType> getClientAuditStrategy() {
89          return interactionId.getWsTransactionConfiguration().getClientAuditStrategy();
90      }
91  
92      @Override
93      public AuditStrategy<AuditDatasetType> getServerAuditStrategy() {
94          return interactionId.getWsTransactionConfiguration().getServerAuditStrategy();
95      }
96  
97      /**
98       * @return static configuration parameters of the Web Service which
99       * server endpoints of this transaction.
100      */
101     public ConfigType getWsTransactionConfiguration() {
102         return interactionId.getWsTransactionConfiguration();
103     }
104 }
