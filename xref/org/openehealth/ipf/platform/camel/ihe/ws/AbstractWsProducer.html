54  /**
55   * Camel producer used to make calls to a Web Service.
56   *
57   * @param <InType>  type of input data (i.e. of the data got from the route).
58   * @param <OutType> type of output data (i.e. of the data returned to the route).
59   * @author Jens Riemschneider
60   * @author Dmytro Rud
61   */
62  public abstract class AbstractWsProducer<
63          AuditDatasetType extends WsAuditDataset,
64          ConfigType extends WsTransactionConfiguration, InType, OutType> extends DefaultProducer {
65      private static final Logger LOG = LoggerFactory.getLogger(AbstractWsProducer.class);
66  
67      private final JaxWsClientFactory<AuditDatasetType> clientFactory;
68      private final Class<InType> requestClass;
69      private final Class<OutType> responseClass;
70  
71  
72      /**
73       * Constructs the producer.
74       *
75       * @param endpoint      the endpoint that creates this producer.
76       * @param clientFactory the factory for clients to produce messages for the service.
77       * @param requestClass  type of request messages.
78       */
79      @SuppressWarnings("unchecked")
80      public AbstractWsProducer(
81              AbstractWsEndpoint<AuditDatasetType, ConfigType> endpoint,
82              JaxWsClientFactory<AuditDatasetType> clientFactory,
83              Class<InType> requestClass,
84              Class<OutType> responseClass) {
85          super(endpoint);
86          notNull(clientFactory, "client factory cannot be null");
87          notNull(requestClass, "request class cannot be null");
88          notNull(responseClass, "responseClass class cannot be null");
89          this.clientFactory = clientFactory;
90          this.requestClass = requestClass;
91          this.responseClass = responseClass;
92      }
93  
94  
95      @Override
96      public void process(Exchange exchange) throws Exception {
97          // prepare
98          InType body = exchange.getIn().getMandatoryBody(requestClass);
99          Object client = getClient();
100         configureClient(client);
101         BindingProvider bindingProvider = (BindingProvider) client;
102         WrappedMessageContext requestContext = (WrappedMessageContext) bindingProvider.getRequestContext();
103         cleanRequestContext(requestContext);
104 
105         enrichRequestContext(exchange, requestContext);
106         processUserDefinedOutgoingHeaders(requestContext, exchange.getIn(), true);
107 
108         // set request encoding based on Camel exchange property
109         String requestEncoding = exchange.getProperty(Exchange.CHARSET_NAME, String.class);
110         if (requestEncoding != null) {
111             requestContext.put(org.apache.cxf.message.Message.ENCODING, requestEncoding);
112         }
113 
114         // get and analyse WS-Addressing asynchrony configuration
115         String replyToUri =
116                 getWsTransactionConfiguration().isAllowAsynchrony()
117                         ? StringUtils.trimToNull(exchange.getIn().getHeader(AbstractWsEndpoint.WSA_REPLYTO_HEADER_NAME, String.class))
118                         : null;
119 
120         // for asynchronous interaction: configure WSA headers and store correlation data
121         if ((replyToUri != null) ||
122                 Boolean.TRUE.equals(requestContext.get(AsynchronyCorrelator.FORCE_CORRELATION))) {
123             String messageId = "urn:uuid:" + UUID.randomUUID().toString();
124             configureWSAHeaders(messageId, replyToUri, requestContext);
125 
126             AsynchronyCorrelator<AuditDatasetType> correlator = getEndpoint().getCorrelator();
127             correlator.storeServiceEndpointUri(messageId, getEndpoint().getEndpointUri());
128 
129             String correlationKey = exchange.getIn().getHeader(
130                     AbstractWsEndpoint.CORRELATION_KEY_HEADER_NAME,
131                     String.class);
132             if (correlationKey != null) {
133                 correlator.storeCorrelationKey(messageId, correlationKey);
134             }
135 
136             String[] alternativeKeys = getAlternativeRequestKeys(exchange);
137             if (alternativeKeys != null) {
138                 correlator.storeAlternativeKeys(messageId, alternativeKeys);
139             }
140         }
141 
142         // invoke
143         exchange.setPattern((replyToUri == null) ? ExchangePattern.InOut : ExchangePattern.InOnly);
144         OutType result = null;
145         try {
146             // normalize response type when called via reflection or similar non-type-safe mechanisms
147             result = responseClass.cast(callService(client, body));
148         } catch (SOAPFaultException fault) {
149             // handle http://www.w3.org/TR/2006/NOTE-soap11-ror-httpbinding-20060321/
150             // see also: https://issues.apache.org/jira/browse/CXF-3768
151             if ((replyToUri == null) ||
152                     (fault.getCause() == null) ||
153                     !(fault.getCause() instanceof WstxEOFException)) {
154                 throw fault;
155             }
156         }
157 
158         // for synchronous interaction (replyToUri == null): handle response.
159         // (async responses are handled in the service instance derived from 
160         // org.openehealth.ipf.platform.camel.ihe.ws.AbstractAsyncResponseWebService)
161         if (replyToUri == null) {
162             Message responseMessage = Exchanges.resultMessage(exchange);
163             responseMessage.getHeaders().putAll(exchange.getIn().getHeaders());
164             WrappedMessageContext responseContext = (WrappedMessageContext) bindingProvider.getResponseContext();
165             processIncomingHeaders(responseContext, responseMessage);
166             enrichResponseMessage(responseMessage, responseContext);
167 
168             // set Camel exchange property based on response encoding
169             exchange.setProperty(Exchange.CHARSET_NAME,
170                     responseContext.get(org.apache.cxf.message.Message.ENCODING));
171             responseMessage.setBody(result, responseClass);
172         }
173     }

221     /**
222      * Sets thread safety & timeout options of the given CXF client.
223      */
224     protected void configureClient(Object o) throws Exception {
225         ClientImpl client = (ClientImpl) ClientProxy.getClient(o);
226         client.setThreadLocalRequestContext(true);
227         client.setSynchronousTimeout(Integer.MAX_VALUE);
228     }
229 
230     /**
231      * Request context is shared among subsequent requests, so we have to clean it.
232      */
233     protected void cleanRequestContext(WrappedMessageContext requestContext) {
234         requestContext.remove(JAXWSAConstants.CLIENT_ADDRESSING_PROPERTIES);
235         requestContext.remove(org.apache.cxf.message.Message.PROTOCOL_HEADERS);
236         requestContext.remove(Header.HEADER_LIST);
237     }

240     /**
241      * Initializes WS-Addressing headers MessageID and ReplyTo,
242      * and stores them into the given message context.
243      */
244     private static void configureWSAHeaders(String messageId, String replyToUri, WrappedMessageContext context) {
245         // obtain headers' container
246         AddressingProperties apropos = (AddressingProperties) context.get(JAXWSAConstants.CLIENT_ADDRESSING_PROPERTIES);
247         if (apropos == null) {
248             apropos = new AddressingProperties();
249             context.put(JAXWSAConstants.CLIENT_ADDRESSING_PROPERTIES, apropos);
250         }
251 
252         // MessageID header
253         AttributedURIType uri = new AttributedURIType();
254         uri.setValue(messageId);
255         apropos.setMessageID(uri);
256         LOG.debug("Set WS-Addressing message ID: {}", messageId);
257 
258         // ReplyTo header
259         if (replyToUri != null) {
260             AttributedURIType uri2 = new AttributedURIType();
261             uri2.setValue(replyToUri);
262             EndpointReferenceType endpointReference = new EndpointReferenceType();
263             endpointReference.setAddress(uri2);
264             apropos.setReplyTo(endpointReference);
265         }
266     }

268     /**
269      * Retrieves the client stub for the Web Service.
270      * <p>
271      * This method caches the client stub instance and therefore requires thread
272      * synchronization.
273      *
274      * @return the client stub.
275      */
276     protected Object getClient() {
277         return clientFactory.getClient(() -> {
278             // Only supply SSLContext if client has not been configured yet
279             SSLContextParameters sslContextParameters = getEndpoint().getSslContextParameters();
280             if (sslContextParameters == null) {
281                 Map<String, SSLContextParameters> sslContextParameterMap = getEndpoint().getCamelContext().getRegistry().findByTypeWithName(SSLContextParameters.class);
282                 if (sslContextParameterMap.size() == 1) {
283                     Map.Entry<String, SSLContextParameters> entry = sslContextParameterMap.entrySet().iterator().next();
284                     sslContextParameters = entry.getValue();
285                 } else if (sslContextParameterMap.size() > 1) {
286                     throw new AmbiguousBeanException(SSLContextParameters.class);
287                 }
288             }
289             try {
290                 SSLContext sslContext = sslContextParameters != null ?
291                         sslContextParameters.createSSLContext(getEndpoint().getCamelContext()) :
292                         null;
293                 return new WsSecurityInformation(getEndpoint().isSecure(), sslContext, getEndpoint().getHostnameVerifier(),
294                         getEndpoint().getUsername(), getEndpoint().getPassword());
295             } catch (Exception e) {
296                 throw new RuntimeCamelException(e);
297             }
298         });
299     }
