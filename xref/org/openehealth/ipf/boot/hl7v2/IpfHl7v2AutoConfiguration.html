62  /**
63   * Configure a basic IPF setup, mostly configuring HL7v2 and Mapping stuff
64   */
65  @Configuration
66  @AutoConfigureAfter(IpfAtnaAutoConfiguration.class)
67  @EnableConfigurationProperties(IpfHl7v2ConfigurationProperties.class)
68  public class IpfHl7v2AutoConfiguration {
69  
70      private static final String IPF_HL7_DEFINITIONS_PREFIX = "org.openehealth.ipf.commons.ihe.hl7v2.definitions";
71  
72      @Autowired
73      private IpfHl7v2ConfigurationProperties properties;
74  
75      @Bean
76      @ConditionalOnMissingBean(HL7MLLPCodec.class)
77      HL7MLLPCodec hl7codec(IpfHl7v2ConfigurationProperties config) {
78          HL7MLLPCodec hl7MLLPCodec = new CustomHL7MLLPCodec();
79          if (config.getCharset() != null) {
80              hl7MLLPCodec.setCharset(config.getCharset());
81          }
82          hl7MLLPCodec.setConvertLFtoCR(config.isConvertLinefeed());
83          return hl7MLLPCodec;
84      }
85  
86      @Bean
87      @ConditionalOnMissingBean(CustomModelClassFactory.class)
88      public CustomModelClassFactory mllpModelClassFactory() {
89          Map<String, String[]> eventMap = new HashMap<>();
90          eventMap.put("2.3.1", new String[] {
91                  IPF_HL7_DEFINITIONS_PREFIX + "pix.v231"
92          });
93          eventMap.put("2.5", new String[] {
94                  IPF_HL7_DEFINITIONS_PREFIX + "pdq.v25",
95                  IPF_HL7_DEFINITIONS_PREFIX + "pix.v25"
96          });
97          CustomModelClassFactory modelClassFactory = new CustomModelClassFactory(eventMap);
98          modelClassFactory.setEventMapDirectory("org/openehealth/ipf/commons/ihe/hl7v2/");
99          return modelClassFactory;
100     }
101 
102     @Bean
103     @ConditionalOnMissingBean(ProfileStore.class)
104     public ProfileStore profileStore() {
105         return new ClasspathProfileStore("/org/openehealth/ipf/gazelle/validation/profile/v2");
106     }
107 
108     @Bean
109     @ConditionalOnMissingBean(ValidationContext.class)
110     public ValidationContext validationContext() {
111         return ValidationContextFactory.noValidation();
112     }
113 
114     @Bean
115     @ConditionalOnMissingBean(IDGenerator.class)
116     @ConditionalOnProperty(prefix = "ipf.hl7v2", name = "generator", havingValue = "file", matchIfMissing = true)
117     public IDGenerator fileGenerator() {
118         return new IpfHiLoIdGenerator(properties.getIdGenerator());
119     }
120 
121     @Bean
122     @ConditionalOnMissingBean(IDGenerator.class)
123     @ConditionalOnProperty(prefix = "ipf.hl7v2", name = "generator", havingValue = "uuid")
124     public IDGenerator uuidGenerator() {
125         return new UUIDGenerator();
126     }
127 
128     @Bean
129     @ConditionalOnMissingBean(IDGenerator.class)
130     @ConditionalOnProperty(prefix = "ipf.hl7v2", name = "generator", havingValue = "nano")
131     public IDGenerator nanoGenerator() {
132         return new NanoTimeGenerator();
133     }
134 
135     @ConfigurationProperties(prefix = "ipf.hl7v2.parser")
136     @Bean
137     public ParserConfiguration parserConfiguration() {
138         return new ParserConfiguration();
139     }
140 
141     @Bean
142     @ConditionalOnMissingBean(HapiContext.class)
143     public HapiContext hapiContext(CustomModelClassFactory modelClassFactory, ProfileStore profileStore,
144                                    ValidationContext validationContext, ParserConfiguration parserConfiguration,
145                                    IDGenerator idGenerator) {
146         HapiContext context = new DefaultHapiContext();
147         context.setModelClassFactory(modelClassFactory);
148         context.setValidationContext(validationContext);
149         context.setProfileStore(profileStore);
150         context.setParserConfiguration(parserConfiguration);
151         context.getParserConfiguration().setIdGenerator(idGenerator);
152         return context;
153     }
154 
155     // Provide bean for MLLP endpoint dispatching
156     @Bean
157     @ConditionalOnMissingBean(ConsumerDispatchingInterceptor.class)
158     public ConsumerDispatchingInterceptor mllpDispatcher(CamelContext camelContext) {
159         return new ConsumerDispatchingInterceptor(camelContext);
160     }
161 
162     // Provide "interactiveContinuationStorage" for HL7v2 paging
163     @Bean
164     @ConditionalOnMissingBean(InteractiveContinuationStorage.class)
165     @ConditionalOnSingleCandidate(CacheManager.class)
166     @ConditionalOnProperty("ipf.hl7v2.caching")
167     public InteractiveContinuationStorage interactiveContinuationStorage(CacheManager cacheManager) {
168         return new CachingInteractiveHl7v2ContinuationStorage(cacheManager);
169     }
170 
171     // Provide "unsolicitedFragmentationStorage" for HL7v2 fragmentation
172     @Bean
173     @ConditionalOnMissingBean(UnsolicitedFragmentationStorage.class)
174     @ConditionalOnSingleCandidate(CacheManager.class)
175     @ConditionalOnProperty("ipf.hl7v2.caching")
176     public UnsolicitedFragmentationStorage unsolicitedFragmentationStorage(CacheManager cacheManager) {
177         return new CachingUnsolicitedFragmentionStorage(cacheManager);
178     }
