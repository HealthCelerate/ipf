40  /**
41   * A flow purge job implementation used by both, Quartz scheduler and
42   * {@link FlowPurgerMBean}. The interface to the Quartz scheduler is
43   * {@link Job} whereas the {@link FlowPurgerMBean} uses this class directly for
44   * managing jobs and and tracking their status.
45   * 
46   * @author Martin Krasser
47   */
48  public class FlowPurgeJob implements Job {
49      
50      private static final String FLOW_MANAGER_KEY = "flowManager";
51      private static final String APP_CONFIG_KEY = "appConfig";
52      private static final Logger LOG = LoggerFactory.getLogger(FlowPurgeJob.class);
53      
54      private final FlowManager flowManager;
55      private final Scheduler scheduler;
56      private final String application;
57  
58      private String flowPurgeSchedule;
59      private String purgeFlowsOlderThan;
60      private boolean doNotPurgeErrorFlows;
61      
62      /**
63       * Constructor used by Quartz.
64       */
65      public FlowPurgeJob() {
66          this(null, null, null);
67      }
68  
69      /**
70       * Constructor used by {@link FlowPurgerMBean}.
71       * 
72       * @param flowManager
73       *            flow manager.
74       * @param scheduler
75       *            Quartz scheduler.
76       * @param application
77       *            application name.
78       */
79      public FlowPurgeJob(FlowManager flowManager, Scheduler scheduler, String application) {
80          this.flowManager = flowManager;
81          this.scheduler = scheduler;
82          this.application = application;
83          flowPurgeSchedule = FLOW_PURGE_SCHEDULE_DEFAULT;
84          purgeFlowsOlderThan = PURGE_FLOWS_OLDER_THAN_DEFAULT;
85          doNotPurgeErrorFlows = DO_NOT_PURGE_ERROR_FLOWS_DEFAULT;
86      }
87      
88      public String getFlowPurgeSchedule() {
89          return flowPurgeSchedule;
90      }
91  
92      /**
93       * @see ApplicationConfig#setFlowPurgeSchedule(String)
94       */
95      public void setFlowPurgeSchedule(String flowPurgeSchedule) {
96          this.flowPurgeSchedule = flowPurgeSchedule;
97      }
98  
99      public String getPurgeFlowsOlderThan() {
100         return purgeFlowsOlderThan;
101     }
102 
103     /**
104      * @see ApplicationConfig#setPurgeFlowsOlderThan(String)
105      */
106     public void setPurgeFlowsOlderThan(String purgeFlowsOlderThan) {
107         this.purgeFlowsOlderThan = purgeFlowsOlderThan;
108     }
109 
110     public boolean isDoNotPurgeErrorFlows() {
111         return doNotPurgeErrorFlows;
112     }
113 
114     /**
115      * @see ApplicationConfig#setDoNotPurgeErrorFlows(boolean) 
116      */
117     public void setDoNotPurgeErrorFlows(boolean doNotPurgeErrorFlows) {
118         this.doNotPurgeErrorFlows = doNotPurgeErrorFlows;
119     }
120 
121     /**
122      * @see org.quartz.Job#execute(org.quartz.JobExecutionContext)
123      */
124     @Override
125     public void execute(JobExecutionContext context) throws JobExecutionException {
126         FlowManager manager = (FlowManager)context.getJobDetail().getJobDataMap().get(FLOW_MANAGER_KEY);
127         ApplicationConfig config = (ApplicationConfig)context.getJobDetail().getJobDataMap().get(APP_CONFIG_KEY);
128         synchronized (manager) {
129             execute(manager, config);
130         }
131     }
132 
133     /**
134      * Runs repeated {@link FlowManager#purgeFlows(FlowPurgeCriteria)}
135      * operations based on criteria given in <code>config</code> until all flows
136      * matching the criteria have been purged from the database. .
137      * 
138      * @param manager
139      *            flow manager
140      * @param config
141      *            application-specific purge configuration
142      */
143     private void execute(FlowManager manager, ApplicationConfig config) {
144         PurgeMode mode = config.isDoNotPurgeErrorFlows() ? CLEAN : ALL;
145         Date olderThan = Duration.parse(config.getPurgeFlowsOlderThan()).since();
146         FlowPurgeCriteria criteria = new FlowPurgeCriteria(mode, olderThan, config.getApplication(), 100);
147     
148         int purgeCountTotal = 0;
149         int purgeCount;
150         LOG.info("Start purging flows for application {}", config.getApplication());
151         do {
152             purgeCount = manager.purgeFlows(criteria);
153             purgeCountTotal += purgeCount;
154             LOG.info("Purged {} flows (application={})", purgeCountTotal, config.getApplication());
155         } while (purgeCount >= criteria.getMaxPurgeCount()); 
156         LOG.info("Finished purging flows for application {}", config.getApplication());
157     }
158 
159     /**
160      * Schedules this job for an single, immediate execution in a separate
161      * thread.
162      * 
163      * @param config
164      *            application-specific purge configuration
165      */
166     public void execute(ApplicationConfig config) {
167         String name = "once";
168 
169         schedule(config, newTrigger().withIdentity(name, null).build(), name);
170         LOG.info("Execute purge job once for application {}", application);
171     }
172     
173     /**
174      * Schedules this job for a repeated execution based on a cron expression.
175      * 
176      * @param config
177      *            application-specific purge configuration
178      * @see ApplicationConfig#setFlowPurgeSchedule(String)
179      */
180     public void schedule(ApplicationConfig config) {
181         CronTrigger cronTrigger = newTrigger().withIdentity(application, null)
182                                               .withSchedule(cronSchedule(flowPurgeSchedule))
183                                               .build();
184         schedule(config, cronTrigger, application);
185         setFlowPurgeScheduled(true);
186         LOG.info("Scheduled purge job for application {}", application);
187     }
188     
189     private void schedule(ApplicationConfig config, Trigger trigger, String jobName) {
190         JobDataMap jobDataMap = new JobDataMap();
191         jobDataMap.put(FlowPurgeJob.FLOW_MANAGER_KEY, flowManager);
192         jobDataMap.put(FlowPurgeJob.APP_CONFIG_KEY, config);
193 
194         JobDetail jobDetail = newJob()
195                                   .withIdentity(jobName, null)
196                                   .ofType(FlowPurgeJob.class)
197                                   .usingJobData(jobDataMap)
198                                   .build();
199         
200         try {
201             scheduler.scheduleJob(jobDetail, trigger);
202         } catch (SchedulerException e) {
203             throw new FlowPurgeJobException("Cannot schedule job: " + e.getMessage());
204         }
205     }
206     
207     /**
208      * Unschedules this job from repeated execution.
209      */
210     public void unschedule() {
211         try {
212             scheduler.deleteJob(JobKey.jobKey(application, null));
213             setFlowPurgeScheduled(false);
214             LOG.info("Unscheduled purge job for application {}", application);
215         } catch (SchedulerException e) {
216             throw new FlowPurgeJobException("Cannot unschedule job: " + e.getMessage());
217         }
218     }
219 
220     /**
221      * Returns <code>true</code> is this job is currently scheduled for repeated
222      * execution. This method does <b>not</b> return
223      * <code> if the job currently as a result of a {@link #execute(ApplicationConfig)} invocation.
224      * 
225      * @return <code>true</code> if the job is scheduled for repeated execution,
226      *         <code>false</code> otherwise.
227      */
228     public boolean isScheduled() {
229         try {
230             return scheduler.getJobDetail(JobKey.jobKey(application, null)) != null;
231         } catch (SchedulerException e) {
232             throw new FlowPurgeJobException("Cannot get job details: " + e.getMessage());
233         }
234     }
235     
236     private void setFlowPurgeScheduled(boolean purgeScheduled) {
237         ApplicationConfig applicationConfig = flowManager.getApplicationConfig(application);
238         applicationConfig.setFlowPurgeScheduled(purgeScheduled);
239         flowManager.mergeApplicationConfig(applicationConfig);
240     }
