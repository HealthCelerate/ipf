16  package org.openehealth.ipf.commons.flow.domain;
17  
18  import java.util.Date;
19  import java.util.UUID;
20  
21  import javax.persistence.Column;
22  import javax.persistence.Entity;
23  import javax.persistence.FetchType;
24  import javax.persistence.Id;
25  import javax.persistence.JoinColumn;
26  import javax.persistence.OneToOne;
27  import javax.persistence.Table;
28  
29  import org.hibernate.annotations.Cascade;
30  import org.hibernate.annotations.CascadeType;
31  import org.hibernate.search.annotations.DocumentId;
32  import org.hibernate.search.annotations.Indexed;
33  import org.hibernate.search.annotations.IndexedEmbedded;
34  import org.openehealth.ipf.commons.flow.transfer.FlowPartInfo;
35  
36  /**
37   * @author Martin Krasser
38   * @author Mitko Kolev 
39   */
40  @Indexed(index="messages.idx")
41  @Entity
42  @Table(name = "T_FLOW_PART")
43  public class FlowPart {
44  
45      @DocumentId
46      @Id
47      @Column(name="C_ID", length=128)
48      private final String identifier; // internal
49      
50      @Column(name="C_STATUS")
51      private FlowStatus status;
52  
53      @Column(name="C_PATH")
54      private String path;
55      
56      @Column(name="C_CONTRIBUTION_TIME")
57      private Date contributionTime;
58      
59      @Column(name="C_CONTRIBUTION_COUNT")
60      private int contributionCount;
61      
62      @Column(name="C_FILTER_TIME")
63      private Date filterTime;
64      
65      @Column(name="C_FILTER_COUNT")
66      private int filterCount;
67      
68      @Column(name="C_FLOW_ID")
69      private Long flowId;
70      
71      @IndexedEmbedded(depth = 1)
72      @OneToOne(fetch=FetchType.LAZY)
73      @JoinColumn(name="FLOW_PART_MESSAGE_ID", unique=false, nullable=true, updatable=true)
74      @Cascade({CascadeType.ALL})
75      private FlowPartMessage flowPartMessage;
76      
77      
78      
79      public FlowPart() {
80          identifier = UUID.randomUUID().toString();
81          contributionCount = 0;
82          filterCount = 0;
83      }
84  
85      public Long getFlowId() {
86          return flowId;
87      }
88      
89      public FlowStatus getStatus() {
90          return status;
91      }
92  
93      public void setStatus(FlowStatus status) {
94          this.status = status;
95      }
96      
97      public String getPath() {
98          return path;
99      }
100 
101     public void setPath(String branch) {
102         path = branch;
103     }
104 
105     public Date getContributionTime() {
106         return contributionTime;
107     }
108 
109     public void setContributionTime(Date contributionTime) {
110         this.contributionTime = contributionTime;
111     }
112     
113     public int getContributionCount() {
114         return contributionCount;
115     }
116 
117     public void setContributionCount(int contributionCount) {
118         this.contributionCount = contributionCount;
119     }
120 
121     public Date getFilterTime() {
122         return filterTime;
123     }
124 
125     public void setFilterTime(Date filterTime) {
126         this.filterTime = filterTime;
127     }
128 
129     public int getFilterCount() {
130         return filterCount;
131     }
132 
133     public void setFilterCount(int filterCount) {
134         this.filterCount = filterCount;
135     }
136 
137     public void incrementContributionCount() {
138         contributionCount++;
139     }
140     
141     public void incrementFilterCount() {
142         filterCount++;
143     }
144     /**
145      * 
146      * Reads the readable text associated with the flow packet from the
147      * database.
148      * 
149      * @return The readable text of the flow packet. If such does not exist,
150      *         returns null.
151      */
152     public String getFlowPartMessageText() {
153         if (flowPartMessage != null) {
154             return flowPartMessage.getText();
155         }
156         return null;
157     }
158 
159     FlowPartMessage getFlowPartMessage() {
160         return flowPartMessage;
161     }
162     
163     /**
164      * Sets the readable text representation of the flow packet. If the message
165      * is null, after the current transaction is committed the existing message
166      * in the database (if such exists) will be deleted.
167      * 
168      * @param message
169      *            the text associated with the packet of this flow.
170      */
171     public void setFlowPartMessageText(String message) {
172         if (message == null) {
173             flowPartMessage = null;
174         } else {
175             flowPartMessage = new FlowPartMessage(message);
176         }
177     }
178     
179     public Date getLatestUpdate() {
180         if (contributionTime == null) {
181             return filterTime;
182         } else if (filterTime == null) {
183             return contributionTime;
184         } else if (filterTime.after(contributionTime)) {
185             return filterTime;
186         } else {
187             return contributionTime;
188         }
189     }
190     
191     public FlowPartInfo getInfo(Flow owner) {
192         return getInfo(owner, false);
193     }
194     
195     public FlowPartInfo getInfo(Flow owner, boolean includeText) {
196         FlowPartInfo info = new FlowPartInfo();
197         info.setStatus(status.toString());
198         info.setPath(path);
199         info.setPathDuration(owner.getPartDuration(this));
200         info.setContributionTime(contributionTime);
201         info.setContributionCount(contributionCount);
202         info.setFilterTime(filterTime);
203         info.setFilterCount(filterCount);
204         if (includeText) {
205             info.setText(getFlowPartMessageText());
206         }
207         return info;
208     }
209     
210     @Override
211     public boolean equals(Object obj) {
212         if (obj == this) {
213             return true;
214         }
215         if (!(obj instanceof FlowPart)) {
216             return false;
217         }
218         FlowPart p = (FlowPart)obj;
219         return identifier.equals(p.identifier);
220     }
221 
222     @Override
223     public int hashCode() {
224         return identifier.hashCode();
225     }
226 
227 }
