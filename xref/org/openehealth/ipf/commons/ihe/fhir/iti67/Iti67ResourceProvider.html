37  /**
38   * Resource Provider for MHD (ITI-67).
39   * <p>
40   * Note (CP by Rick Riemer):
41   * When searching for XDS documents with specific referenceIdList values, MHD specifies to use the related-id query parameter.
42   * This parameter is of type token (https://www.hl7.org/fhir/search.html#token). The token type does not allow searching for
43   * the Identifier.type attribute, which would be a primary use case.
44   * <p>
45   * IHE should provide a mechanism to search for referenceIdList values by type, in addition to system and value.
46   * Suggestion: donâ€™t use token, but use composite, (https://www.hl7.org/fhir/search.html#composite) and define how
47   * to use it for searching against referenceIdList values. A composite parameter could look like:
48   * related id=urn:oid:1.2.3.4.5.6|2013001$urn:ihe:iti:xds:2013:accession.
49   *
50   * @author Christian Ohr
51   * @since 3.2
52   */
53  public class Iti67ResourceProvider extends AbstractPlainProvider {
54  
55      @SuppressWarnings("unused")
56      @Search(type = DocumentReference.class)
57      public IBundleProvider documentManifestSearch(
58              @RequiredParam(name = DocumentReference.SP_PATIENT, chainWhitelist = {"", Patient.SP_IDENTIFIER}) ReferenceParam patient,
59              @OptionalParam(name = DocumentReference.SP_INDEXED) DateRangeParam indexed,
60              @OptionalParam(name = DocumentReference.SP_AUTHOR + "." + Practitioner.SP_FAMILY) StringParam authorFamilyName,
61              @OptionalParam(name = DocumentReference.SP_AUTHOR + "." + Practitioner.SP_GIVEN) StringParam authorGivenName,
62              @OptionalParam(name = DocumentReference.SP_STATUS) TokenOrListParam status,
63              @OptionalParam(name = DocumentReference.SP_CLASS) TokenOrListParam class_,
64              @OptionalParam(name = DocumentReference.SP_TYPE) TokenOrListParam type,
65              @OptionalParam(name = DocumentReference.SP_SETTING) TokenOrListParam setting,
66              @OptionalParam(name = DocumentReference.SP_PERIOD) DateRangeParam period,
67              @OptionalParam(name = DocumentReference.SP_FACILITY) TokenOrListParam facility,
68              @OptionalParam(name = DocumentReference.SP_EVENT) TokenOrListParam event,
69              @OptionalParam(name = DocumentReference.SP_SECURITYLABEL) TokenOrListParam securityLabel,
70              @OptionalParam(name = DocumentReference.SP_FORMAT) TokenOrListParam format,
71              @OptionalParam(name = DocumentReference.SP_RELATEDID) TokenOrListParam relatedId,
72              // Extension to ITI-66
73              @OptionalParam(name = IAnyResource.SP_RES_ID) TokenParam resourceId,
74              @Sort SortSpec sortSpec,
75              HttpServletRequest httpServletRequest,
76              HttpServletResponse httpServletResponse) {
77  
78          Iti67SearchParameters searchParameters = Iti67SearchParameters.builder()
79                  .indexed(indexed)
80                  .authorFamilyName(authorFamilyName)
81                  .authorGivenName(authorGivenName)
82                  .status(status)
83                  .class_(class_)
84                  .type(type)
85                  .setting(setting)
86                  .period(period)
87                  .facility(facility)
88                  .event(event)
89                  .securityLabel(securityLabel)
90                  .format(format)
91                  .relatedId(relatedId)
92                  ._id(resourceId)
93                  .sortSpec(sortSpec)
94                  .fhirContext(getFhirContext())
95                  .build();
96  
97          String chain = patient.getChain();
98          if (Patient.SP_IDENTIFIER.equals(chain)) {
99              searchParameters.setPatientIdentifier(patient.toTokenParam(getFhirContext()));
100         } else if (chain == null || chain.isEmpty()) {
101             searchParameters.setPatientReference(patient);
102         }
103 
104         // Run down the route
105         return requestBundleProvider(null, searchParameters, httpServletRequest, httpServletResponse);
106     }
107 
108     /**
109      * Handles DocumentReference Retrieve. This is not an actual part of the ITI-67 specification, but in the
110      * context of restful FHIR IHE transaction it makes sense to be able to retrieve a DocumentReference by
111      * its resource ID.
112      *
113      * @param id                  resource ID
114      * @param httpServletRequest  servlet request
115      * @param httpServletResponse servlet response
116      * @return {@link DocumentReference} resource
117      */
118     @SuppressWarnings("unused")
119     @Read(version = true, type = DocumentReference.class)
120     public DocumentReference documentReferenceRetrieve(
121             @IdParam IdType id,
122             HttpServletRequest httpServletRequest,
123             HttpServletResponse httpServletResponse) {
124 
125         // Run down the route
126         return requestResource(id, DocumentReference.class, httpServletRequest, httpServletResponse);
127     }
