33  /**
34   * Transforms between a {@link DocumentEntry} and its ebXML representation.
35   * @author Jens Riemschneider
36   */
37  public class DocumentEntryTransformer extends XDSMetaClassTransformer<EbXMLExtrinsicObject, DocumentEntry> {
38      private final EbXMLFactory factory;
39      
40      private final AuthorTransformer authorTransformer;
41      private final CodeTransformer codeTransformer;
42      
43      private final PatientInfoTransformer patientInfoTransformer = new PatientInfoTransformer();
44      private final UriTransformer uriTransformer = new UriTransformer();   
45      
46      /**
47       * Constructs the transformer
48       * @param factory
49       *          factory for version independent ebXML objects. 
50       */
51      public DocumentEntryTransformer(EbXMLFactory factory) {
52          super(DOC_ENTRY_PATIENT_ID_EXTERNAL_ID, 
53                  DOC_ENTRY_LOCALIZED_STRING_PATIENT_ID, 
54                  DOC_ENTRY_UNIQUE_ID_EXTERNAL_ID,
55                  DOC_ENTRY_LOCALIZED_STRING_UNIQUE_ID);
56          
57          notNull(factory, "factory cannot be null");
58  
59          this.factory = factory;
60          authorTransformer = new AuthorTransformer(factory);
61          codeTransformer = new CodeTransformer(factory);
62      }
63      
64      @Override
65      protected EbXMLExtrinsicObject createEbXMLInstance(String id, EbXMLObjectLibrary objectLibrary) {
66          return factory.createExtrinsic(id, objectLibrary);
67      }
68      
69      @Override
70      protected DocumentEntry createMetaClassInstance() {
71          return new DocumentEntry();
72      }
73  
74      @Override
75      protected void addAttributesFromEbXML(DocumentEntry docEntry, EbXMLExtrinsicObject extrinsic) {
76          super.addAttributesFromEbXML(docEntry, extrinsic);
77          docEntry.setAvailabilityStatus(extrinsic.getStatus());        
78          docEntry.setMimeType(extrinsic.getMimeType());
79          docEntry.setType(DocumentEntryType.valueOfUuid(extrinsic.getObjectType()));
80          docEntry.setHomeCommunityId(extrinsic.getHome());
81      }
82  
83      @Override
84      protected void addAttributes(DocumentEntry metaData, EbXMLExtrinsicObject ebXML, EbXMLObjectLibrary objectLibrary) {
85          super.addAttributes(metaData, ebXML, objectLibrary);
86          ebXML.setStatus(metaData.getAvailabilityStatus());                
87          ebXML.setMimeType(metaData.getMimeType());
88          ebXML.setObjectType(DocumentEntryType.toUuid(metaData.getType()));
89          ebXML.setHome(metaData.getHomeCommunityId());
90      }
91  
92      @Override
93      protected void addSlotsFromEbXML(DocumentEntry docEntry, EbXMLExtrinsicObject extrinsic) {
94          super.addSlotsFromEbXML(docEntry, extrinsic);
95          
96          docEntry.setCreationTime(extrinsic.getSingleSlotValue(SLOT_NAME_CREATION_TIME));
97          docEntry.setHash(extrinsic.getSingleSlotValue(SLOT_NAME_HASH));
98          docEntry.setLanguageCode(extrinsic.getSingleSlotValue(SLOT_NAME_LANGUAGE_CODE));
99          docEntry.setServiceStartTime(extrinsic.getSingleSlotValue(SLOT_NAME_SERVICE_START_TIME));
100         docEntry.setServiceStopTime(extrinsic.getSingleSlotValue(SLOT_NAME_SERVICE_STOP_TIME));
101         docEntry.setRepositoryUniqueId(extrinsic.getSingleSlotValue(SLOT_NAME_REPOSITORY_UNIQUE_ID));
102         docEntry.setUri(uriTransformer.fromEbXML(extrinsic.getSlotValues(SLOT_NAME_URI)));
103         docEntry.setDocumentAvailability(DocumentAvailability.valueOfOpcode(
104                 extrinsic.getSingleSlotValue(SLOT_NAME_DOCUMENT_AVAILABILITY)));
105         
106         String size = extrinsic.getSingleSlotValue(SLOT_NAME_SIZE);
107         docEntry.setSize(size != null ? Long.parseLong(size) : null);
108         
109         String hl7LegalAuthenticator = extrinsic.getSingleSlotValue(SLOT_NAME_LEGAL_AUTHENTICATOR);
110         docEntry.setLegalAuthenticator(Hl7v2Based.parse(hl7LegalAuthenticator, Person.class));
111         
112         String sourcePatient = extrinsic.getSingleSlotValue(SLOT_NAME_SOURCE_PATIENT_ID);
113         docEntry.setSourcePatientId(Hl7v2Based.parse(sourcePatient, Identifiable.class));
114         
115         List<String> slotValues = extrinsic.getSlotValues(SLOT_NAME_SOURCE_PATIENT_INFO);        
116         docEntry.setSourcePatientInfo(patientInfoTransformer.fromHL7(slotValues));
117 
118         for (String referenceIdValue : extrinsic.getSlotValues(SLOT_NAME_REFERENCE_ID_LIST)) {
119             docEntry.getReferenceIdList().add(Hl7v2Based.parse(referenceIdValue, ReferenceId.class));
120         }
121     }
122     
123     @Override
124     protected void addSlots(DocumentEntry docEntry, EbXMLExtrinsicObject extrinsic, EbXMLObjectLibrary objectLibrary) {
125         super.addSlots(docEntry, extrinsic, objectLibrary);
126         
127         extrinsic.addSlot(SLOT_NAME_CREATION_TIME, toHL7(docEntry.getCreationTime()));
128         extrinsic.addSlot(SLOT_NAME_HASH, docEntry.getHash());
129         extrinsic.addSlot(SLOT_NAME_LANGUAGE_CODE, docEntry.getLanguageCode());
130         extrinsic.addSlot(SLOT_NAME_SERVICE_START_TIME, toHL7(docEntry.getServiceStartTime()));
131         extrinsic.addSlot(SLOT_NAME_SERVICE_STOP_TIME, toHL7(docEntry.getServiceStopTime()));
132         extrinsic.addSlot(SLOT_NAME_REPOSITORY_UNIQUE_ID, docEntry.getRepositoryUniqueId());
133         extrinsic.addSlot(SLOT_NAME_URI, uriTransformer.toEbXML(docEntry.getUri()));
134         extrinsic.addSlot(SLOT_NAME_DOCUMENT_AVAILABILITY,
135                 DocumentAvailability.toFullQualifiedOpcode(docEntry.getDocumentAvailability()));
136         
137         Long size = docEntry.getSize();
138         extrinsic.addSlot(SLOT_NAME_SIZE, size != null ? size.toString() : null);
139         
140         String hl7LegalAuthenticator = Hl7v2Based.render(docEntry.getLegalAuthenticator());
141         extrinsic.addSlot(SLOT_NAME_LEGAL_AUTHENTICATOR, hl7LegalAuthenticator);
142         
143         String sourcePatient = Hl7v2Based.render(docEntry.getSourcePatientId());
144         extrinsic.addSlot(SLOT_NAME_SOURCE_PATIENT_ID, sourcePatient);
145         
146         List<String> slotValues = patientInfoTransformer.toHL7(docEntry.getSourcePatientInfo());
147         extrinsic.addSlot(SLOT_NAME_SOURCE_PATIENT_INFO, slotValues.toArray(new String[slotValues.size()]));
148 
149         if (! docEntry.getReferenceIdList().isEmpty()) {
150             String[] referenceIdValues = new String[docEntry.getReferenceIdList().size()];
151             for (int i = 0; i < docEntry.getReferenceIdList().size(); ++i) {
152                 referenceIdValues[i] = Hl7v2Based.render(docEntry.getReferenceIdList().get(i));
153             }
154             extrinsic.addSlot(SLOT_NAME_REFERENCE_ID_LIST, referenceIdValues);
155         }
156     }
157 
158     @Override
159     protected void addClassificationsFromEbXML(DocumentEntry docEntry, EbXMLExtrinsicObject extrinsic) {
160         super.addClassificationsFromEbXML(docEntry, extrinsic);
161 
162         for (EbXMLClassification author : extrinsic.getClassifications(DOC_ENTRY_AUTHOR_CLASS_SCHEME)) {
163             docEntry.getAuthors().add(authorTransformer.fromEbXML(author));
164         }
165         
166         EbXMLClassification classCode = extrinsic.getSingleClassification(DOC_ENTRY_CLASS_CODE_CLASS_SCHEME);
167         docEntry.setClassCode(codeTransformer.fromEbXML(classCode));
168 
169         EbXMLClassification formatCode = extrinsic.getSingleClassification(DOC_ENTRY_FORMAT_CODE_CLASS_SCHEME);
170         docEntry.setFormatCode(codeTransformer.fromEbXML(formatCode));
171 
172         EbXMLClassification healthcareFacility = extrinsic.getSingleClassification(DOC_ENTRY_HEALTHCARE_FACILITY_TYPE_CODE_CLASS_SCHEME);
173         docEntry.setHealthcareFacilityTypeCode(codeTransformer.fromEbXML(healthcareFacility));
174         
175         EbXMLClassification practiceSetting = extrinsic.getSingleClassification(DOC_ENTRY_PRACTICE_SETTING_CODE_CLASS_SCHEME);
176         docEntry.setPracticeSettingCode(codeTransformer.fromEbXML(practiceSetting));
177         
178         EbXMLClassification typeCode = extrinsic.getSingleClassification(DOC_ENTRY_TYPE_CODE_CLASS_SCHEME);
179         docEntry.setTypeCode(codeTransformer.fromEbXML(typeCode));
180         
181         List<Code> confidentialityCodes = docEntry.getConfidentialityCodes();
182         confidentialityCodes.addAll(extrinsic.getClassifications(DOC_ENTRY_CONFIDENTIALITY_CODE_CLASS_SCHEME).stream()
183                 .map(codeTransformer::fromEbXML)
184                 .collect(Collectors.toList()));
185 
186         List<Code> eventCodeList = docEntry.getEventCodeList();
187         eventCodeList.addAll(extrinsic.getClassifications(DOC_ENTRY_EVENT_CODE_CLASS_SCHEME).stream()
188                 .map(codeTransformer::fromEbXML)
189                 .collect(Collectors.toList()));
190 
191         List<EbXMLClassification> limitedMetadata = extrinsic.getClassifications(DOC_ENTRY_LIMITED_METADATA_CLASS_SCHEME);
192         docEntry.setLimitedMetadata(! limitedMetadata.isEmpty());
193     }
194 
195     @Override
196     protected void addClassifications(DocumentEntry docEntry, EbXMLExtrinsicObject extrinsic, EbXMLObjectLibrary objectLibrary) {
197         super.addClassifications(docEntry, extrinsic, objectLibrary);
198         
199         for (Author author : docEntry.getAuthors()) {
200             EbXMLClassification authorClasification = authorTransformer.toEbXML(author, objectLibrary);
201             extrinsic.addClassification(authorClasification, DOC_ENTRY_AUTHOR_CLASS_SCHEME);
202         }
203         
204         EbXMLClassification classCode = codeTransformer.toEbXML(docEntry.getClassCode(), objectLibrary);
205         extrinsic.addClassification(classCode, DOC_ENTRY_CLASS_CODE_CLASS_SCHEME);
206         
207         EbXMLClassification formatCode = codeTransformer.toEbXML(docEntry.getFormatCode(), objectLibrary);
208         extrinsic.addClassification(formatCode, DOC_ENTRY_FORMAT_CODE_CLASS_SCHEME);
209 
210         EbXMLClassification healthcareFacility = codeTransformer.toEbXML(docEntry.getHealthcareFacilityTypeCode(), objectLibrary);
211         extrinsic.addClassification(healthcareFacility, DOC_ENTRY_HEALTHCARE_FACILITY_TYPE_CODE_CLASS_SCHEME);
212         
213         EbXMLClassification practiceSetting = codeTransformer.toEbXML(docEntry.getPracticeSettingCode(), objectLibrary);
214         extrinsic.addClassification(practiceSetting, DOC_ENTRY_PRACTICE_SETTING_CODE_CLASS_SCHEME);
215         
216         EbXMLClassification typeCode = codeTransformer.toEbXML(docEntry.getTypeCode(), objectLibrary);
217         extrinsic.addClassification(typeCode, DOC_ENTRY_TYPE_CODE_CLASS_SCHEME);
218         
219         for (Code confCode : docEntry.getConfidentialityCodes()) {
220             EbXMLClassification conf = codeTransformer.toEbXML(confCode, objectLibrary);
221             extrinsic.addClassification(conf, DOC_ENTRY_CONFIDENTIALITY_CODE_CLASS_SCHEME);
222         }
223         
224         for (Code eventCode : docEntry.getEventCodeList()) {
225             EbXMLClassification event = codeTransformer.toEbXML(eventCode, objectLibrary);
226             extrinsic.addClassification(event, DOC_ENTRY_EVENT_CODE_CLASS_SCHEME);
227         }
228 
229         if (docEntry.isLimitedMetadata()) {
230             EbXMLClassification classification = factory.createClassification(objectLibrary);
231             extrinsic.addClassification(classification, DOC_ENTRY_LIMITED_METADATA_CLASS_SCHEME);
232         }
233     }
