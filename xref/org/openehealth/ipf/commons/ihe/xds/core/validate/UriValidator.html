16  package org.openehealth.ipf.commons.ihe.xds.core.validate;
17  
18  import static org.openehealth.ipf.commons.ihe.xds.core.validate.ValidationMessage.*;
19  import static org.openehealth.ipf.commons.ihe.xds.core.validate.ValidatorAssertions.*;
20  
21  import java.util.List;
22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  
25  import java.net.MalformedURLException;
26  import java.net.URI;
27  import java.net.URISyntaxException;
28  import java.net.URL;
29  
30  /**
31   * Validates a value list for compliance with a URI (RFC 2616).
32   * @author Jens Riemschneider
33   */
34  public class UriValidator implements ValueListValidator {
35      private static final Pattern PATTERN = Pattern.compile("([1-9])\\|(.*)");
36  
37      @Override
38      public void validate(List<String> values) throws XDSMetaDataException {
39          if (values.size() > 0) {
40              String uri = getUri(values);
41              metaDataAssert(uri != null, NULL_URI);
42              metaDataAssert(!uri.isEmpty(), EMPTY_URI);
43              
44              // Accept anything that the classes URI or URL accept. This is done to
45              // avoid e.g. "http://" to fail. The XDSToolKit is using this URI
46              // for some tests and the RFCs do not clearly state if this is a valid
47              // URI or not. The URL class seems to accept it, the URI class doesn't.
48              try {
49                  new URI(uri);
50              }
51              catch (URISyntaxException eUri) {
52                  try {
53                      new URL(uri);
54                  } catch (MalformedURLException eUrl) {
55                      throw new XDSMetaDataException(INVALID_URI, uri);
56                  }
57              }
58          }
59      }
60      
61      private String getUri(List<String> values) throws XDSMetaDataException {
62          if (values.size() == 1) {
63              String uri = values.get(0);
64              if (uri == null) {
65                  return null;
66              }
67              
68              if (uri.indexOf('|') == -1) {
69                  return uri;
70              }
71          }
72          
73          String[] uriParts = new String[10];
74          int highestIdx = 0;
75          for (String slotValue : values) {
76              metaDataAssert(slotValue != null, NULL_URI_PART);
77              
78              Matcher matcher = PATTERN.matcher(slotValue);
79              metaDataAssert(matcher.matches(), INVALID_URI_PART, slotValue);
80              
81              int uriIdx = Integer.parseInt(matcher.group(1));
82              uriParts[uriIdx] = matcher.group(2);
83              highestIdx = Math.max(highestIdx, uriIdx);
84          }
85          
86          StringBuilder builder = new StringBuilder();
87          for (int idx = 1; idx <= highestIdx; ++idx) {
88              String uriPart = uriParts[idx];
89              metaDataAssert(uriPart != null, MISSING_URI_PART, idx);
90              builder.append(uriPart);
91          }
92          
93          return builder.toString();
94      }
95  }
