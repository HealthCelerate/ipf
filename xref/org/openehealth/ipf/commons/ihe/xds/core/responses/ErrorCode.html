168 
169     private final String opcode;
170     
171     ErrorCode(String opcode) {
172         this.opcode = opcode;
173     }
174 
175     /**
176      * @return string representation for usage in ebXML values.
177      */
178     public String getOpcode() {
179         return opcode;
180     }
181 
182     /**
183      * <code>null</code>-safe version of {@link #getOpcode()}.
184      * @param errorCode
185      *          the error code. Can be <code>null</code>.
186      * @return the string representation or <code>null</code> if errorCode was <code>null</code>.
187      */
188     public static String getOpcode(ErrorCode errorCode) {
189         return errorCode != null ? errorCode.getOpcode() : null;
190     }
191  
192     /**
193      * Returns the error code that corresponds to the given opcode.
194      * @param opcode
195      *          the opcode. Can be <code>null</code>.
196      * @return the error code.
197      *      <code>null</code> when the opcode is <code>null</code> or empty,
198      *      {@link #_USER_DEFINED} when the code is not a standard one.
199      */
200     public static ErrorCode valueOfOpcode(String opcode) {
201         if (StringUtils.isBlank(opcode)) {
202             return null;
203         }
204         
205         for (ErrorCode code : values()) {
206             if (opcode.equals(code.getOpcode())) {
207                 return code;
208             }
209         }
210 
211         return _USER_DEFINED;
212     }
