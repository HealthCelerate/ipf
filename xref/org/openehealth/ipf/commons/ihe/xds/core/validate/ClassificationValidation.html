16  package org.openehealth.ipf.commons.ihe.xds.core.validate;

18  import org.apache.commons.lang3.Validate;
19  import org.openehealth.ipf.commons.ihe.xds.core.ebxml.EbXMLClassification;
20  import org.openehealth.ipf.commons.ihe.xds.core.ebxml.EbXMLRegistryObject;
21  import org.openehealth.ipf.commons.ihe.xds.core.metadata.Vocabulary.DisplayNameUsage;
22  
23  import java.util.List;
24  
25  import static org.openehealth.ipf.commons.ihe.xds.core.validate.ValidationMessage.*;
26  import static org.openehealth.ipf.commons.ihe.xds.core.validate.ValidatorAssertions.metaDataAssert;
27  
28  /**
29   * Validates a classification.
30   * @author Jens Riemschneider
31   * @author Mitko Kolev
32   */
33  public class ClassificationValidation implements RegistryObjectValidator {
34      protected final String classScheme;
35      private final int min;
36      private final int max;
37      private final DisplayNameUsage displayNameUsage;
38      private final SlotValueValidation[] slotValidations;
39  
40      
41      /**
42       * Constructs the validation with exactly one classification with the given <code>classScheme</code>
43       *  no checks for name.
44       * @param classScheme
45       *          the class scheme of the classification to check.
46       *          the maximum number of classifications allowed for the given scheme.
47       * @param displayNameUsage
48       *          the usage of the display name element.  
49       * @param slotValidations
50       *          validations to apply to the slots of the classification.
51       * @see DisplayNameUsage
52       */
53      public ClassificationValidation(String classScheme, DisplayNameUsage displayNameUsage, SlotValueValidation[] slotValidations) {
54          this (classScheme, 1, 1, displayNameUsage, slotValidations);
55      }
56      
57      /**
58       * Constructs the validation for classifications with the given <code>classScheme</code>.
59       *  
60       * @param classScheme
61       *          the class scheme of the classification to check.
62       * @param min
63       *          the minimum number of classifications allowed for the given scheme.
64       * @param max
65       *          the maximum number of classifications allowed for the given scheme.
66       * @param displayNameUsage
67       *          the usage of the display name element.
68       * @param slotValidations
69       *          validations to apply to the slots of the classification.
70       *          
71       * @see DisplayNameUsage
72       */
73      public ClassificationValidation(String classScheme, int min, int max, DisplayNameUsage displayNameUsage, SlotValueValidation[] slotValidations) {
74          this.classScheme = Validate.notNull(classScheme);
75          this.min = min;
76          this.max = max;
77          this.displayNameUsage = Validate.notNull(displayNameUsage);
78          this.slotValidations = slotValidations;
79      }
80  
81      @Override
82      public void validate(EbXMLRegistryObject obj) throws XDSMetaDataException {
83          List<EbXMLClassification> classifications = obj.getClassifications(classScheme);
84          metaDataAssert(classifications.size() >= min && classifications.size() <= max,
85                  WRONG_NUMBER_OF_CLASSIFICATIONS, classScheme, min, max, classifications.size());
86          for (EbXMLClassification classification : classifications) {
87              metaDataAssert(classification.getClassifiedObject() != null, 
88                      NO_CLASSIFIED_OBJ, classScheme);
89  
90              metaDataAssert(classification.getClassifiedObject().equals(obj.getId()), 
91                      WRONG_CLASSIFIED_OBJ, obj.getId(), classification.getClassifiedObject());
92              
93              metaDataAssert(classification.getNodeRepresentation() != null,
94                      WRONG_NODE_REPRESENTATION, classScheme);
95              
96              switch(displayNameUsage){
97                  case OPTIONAL:
98                      break;
99                  case REQUIRED:
100                     assertDisplayNamePresent(classification, classScheme);
101                     break;
102                 default :
103                     throw new IllegalArgumentException("Unsupported display name usage " + displayNameUsage);
104             }
105             if (slotValidations != null) {
106                 for (SlotValueValidation slotValidation : slotValidations) {
107                     slotValidation.validate(classification);
108                 }
109             }
110         }
111     }
112     
113     public static void assertDisplayNamePresent(EbXMLClassification classification, String classificationScheme) {
114         metaDataAssert(classification.getName() != null, NO_CLASSIFICATION_NAME_OBJ, 
115                 classificationScheme, classification.getClassifiedObject());
116         metaDataAssert(classification.getName().getValue() != null, NO_CLASSIFICATION_NAME_OBJ, 
117                 classificationScheme, classification.getClassifiedObject());
118     }
