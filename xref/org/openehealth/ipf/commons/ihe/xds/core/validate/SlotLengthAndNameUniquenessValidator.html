33  /**
34   * Validates lengths of ebXML slot values and uniqueness of slot names.
35   *
36   * @author Jens Riemschneider
37   * @author Dmytro Rud
38   */
39  public class SlotLengthAndNameUniquenessValidator {
40  
41      /**
42       * Validates slot lengths and name uniqueness
43       * for the given ebXML object container.
44       *
45       * @param container the container of ebXML objects.
46       * @throws XDSMetaDataException if a slot length validation failed.
47       */
48      public void validateContainer(EbXMLObjectContainer container) throws XDSMetaDataException {
49          validateRegistryObjects(container.getAssociations());
50          validateRegistryObjects(container.getExtrinsicObjects());
51          validateRegistryObjects(container.getRegistryPackages());
52          validateSlotLists(container.getClassifications());
53      }
54  
55      private void validateRegistryObjects(List<? extends EbXMLRegistryObject> regObjects) throws XDSMetaDataException {
56          validateSlotLists(regObjects);
57          for (EbXMLRegistryObject regObj : regObjects) {
58              validateSlotLists(regObj.getClassifications());
59          }
60      }
61  
62      private void validateSlotLists(List<? extends EbXMLSlotList> slotListContainers) throws XDSMetaDataException {
63          for (EbXMLSlotList slotList : slotListContainers) {
64              doValidateSlots(slotList.getSlots(), false, null);
65          }
66      }
67  
68      public void validateQuerySlots(
69              List<? extends EbXMLSlot> slots,
70              Set<String> allowedSlotNamesMultiple) throws XDSMetaDataException {
71          doValidateSlots(slots, true, allowedSlotNamesMultiple);
72      }
73  
74      /**
75       * Validates uniqueness of slot names and maximal lengths of slot values in the given collection.
76       *
77       * @param slots                    ebXML slot collection.
78       * @param queryMode                <code>true</code> iff the given slots represent parameters of a stored query.
79       * @param allowedSlotNamesMultiple names of slots which are allowed to be present more than once (only for queries).
80       * @throws XDSMetaDataException when the validation fails.
81       */
82      private void doValidateSlots(
83              List<? extends EbXMLSlot> slots,
84              boolean queryMode,
85              Set<String> allowedSlotNamesMultiple) throws XDSMetaDataException {
86          HashSet<String> names = new HashSet<>();
87          for (EbXMLSlot slot : slots) {
88              // validate format and uniqueness of slot names
89              String name = slot.getName();
90              metaDataAssert(StringUtils.isNotEmpty(name), MISSING_SLOT_NAME);
91  
92              if (queryMode) {
93                  metaDataAssert((name.length() > 1) && (name.charAt(0) == '$'), WRONG_QUERY_SLOT_NAME, name);
94                  metaDataAssert(names.add(name)
95                                  || allowedSlotNamesMultiple.contains(name)
96                                  || isExtraMetadataSlotName(name.substring(1)),
97                          DUPLICATE_SLOT_NAME, name);
98              } else {
99                  metaDataAssert(names.add(name), DUPLICATE_SLOT_NAME, name);
100             }
101 
102             // validate lengths of slot values
103             slot.getValueList().stream()
104                     .filter(value -> value != null)
105                     .forEach(value -> metaDataAssert(value.length() <= slot.getValueLengthLimit(), SLOT_VALUE_TOO_LONG, name));
106         }
107     }
