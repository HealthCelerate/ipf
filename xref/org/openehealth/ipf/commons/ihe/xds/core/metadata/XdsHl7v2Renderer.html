16  package org.openehealth.ipf.commons.ihe.xds.core.metadata;
17  
18  import ca.uhn.hl7v2.HL7Exception;
19  import ca.uhn.hl7v2.model.Composite;
20  import ca.uhn.hl7v2.model.Primitive;
21  import ca.uhn.hl7v2.model.Type;
22  import ca.uhn.hl7v2.model.v25.datatype.*;
23  import ca.uhn.hl7v2.parser.DefaultEscaping;

25  import org.apache.commons.lang3.StringUtils;
26  import java.util.*;

28  /**
29   * A renderer of HL7 v2 elements which considers XDS-specific
30   * requirements regarding required and prohibited fields
31   * as prescribed in the ITI TF Volume 3 Chapter 4.
32   *
33   * @author Dmytro Rud
34   */
35  public abstract class XdsHl7v2Renderer {
36  
37      /**
38       * Encoding characters for HL7 v2 messages.
39       */
40      public static final EncodingCharacters ENCODING_CHARACTERS =
41              new EncodingCharacters('|', '^', '~', '\\', '&');
42  
43      /**
44       * Map from HL7 class name to indices of fields allowed for rendering.
45       * An optional modifier is the class name of the root XDS model object.
46       * So the type of the key is a tuple [HL7 class name, XDS class name].
47       * <p>
48       * Querying algorithm when rendering an HL7 element C which is
49       * contained (directly or indirectly) in an XDS model object T:
50       * <ol>
51       *      <li> Try to retrieve indices for [C class name, null].
52       *      <li> When step 1 fails, try to retrieve indices for [C class name, T class name].
53       *      <li> When step 2 fails, assume that all fields of C are allowed.
54       * </ol>
55       * <p>
56       * Note that the indices are one-based, as in the HL7 and IHE documentation.
57       */
58      private static final Map<String, Collection<Integer>> INCLUSIONS = new HashMap<>();
59  
60      private static void addInclusion(
61              Class<? extends Composite> hl7Class,
62              Class<? extends Hl7v2Based> xdsClass,
63              int... fieldNumbers)
64      {
65          Collection<Integer> collection = new HashSet<>(fieldNumbers.length);
66          for (int number : fieldNumbers) {
67              collection.add(number - 1);
68          }
69          StringBuilder key = new StringBuilder(hl7Class.getSimpleName());
70          if (xdsClass != null) {
71              key.append('\n').append(xdsClass.getSimpleName());
72          }
73          INCLUSIONS.put(key.toString(), collection);
74      }
75  
76      static {
77          addInclusion(CE.class,  null,                        1, 3);
78          addInclusion(CX.class,  Identifiable.class,          1, 4);
79          addInclusion(CX.class,  ReferenceId.class,           1, 4, 5);
80          addInclusion(HD.class,  AssigningAuthority.class,    2, 3);
81          addInclusion(HD.class,  CXiAssigningAuthority.class, 1, 2, 3);
82          addInclusion(HD.class,  Identifiable.class,          2, 3);
83          addInclusion(HD.class,  ReferenceId.class,           1, 2, 3);
84          addInclusion(XON.class, null,                        1, 6, 10);
85          addInclusion(XTN.class, null,                        2, 3, 4, 5, 6, 7, 8);
86      }
87  
88  
89      private XdsHl7v2Renderer() {
90          throw new IllegalStateException("cannot instantiate helper class");
91      }
92  
93      public static boolean isEmpty(Hl7v2Based hl7v2based) {
94          try {
95              return isEmpty(hl7v2based.getHapiObject(), "\n" + hl7v2based.getClass().getSimpleName());
96          } catch (HL7Exception e) {
97              throw new RuntimeException(e);
98          }
99      }
100 
101     private static boolean isEmpty(Composite composite, String keyModifier) throws HL7Exception {
102         Type[] fields = composite.getComponents();
103 
104         String key = composite.getClass().getSimpleName();
105         Collection<Integer> inclusions = INCLUSIONS.get(key);
106         if (inclusions == null) {
107             inclusions = INCLUSIONS.get(key + keyModifier);
108         }
109 
110         for (int i = 0; i < fields.length; ++i) {
111             if ((inclusions == null) || inclusions.contains(i)) {
112                 if (fields[i] instanceof Composite) {
113                     if (!isEmpty((Composite) fields[i], keyModifier)) {
114                         return false;
115                     }
116                 } else if (fields[i] instanceof Primitive) {
117                     if (!fields[i].isEmpty()) {
118                         return false;
119                     }
120                 } else {
121                     // actually, this line should be unreachable
122                     throw new IllegalStateException("Don't know how to handle " + fields[i]);
123                 }
124             }
125         }
126 
127         return true;
128     }
129 
130     /**
131      * Encodes the given HL7-based XDS model object using specific
132      * rules regarding required and prohibited HL7 fields.
133      * @param hl7v2based
134      *      source HL7-based XDS object model to be rendered.
135      * @return
136      *      String representation of the given HL7 v2 composite field.
137      */
138     public static String encode(Hl7v2Based hl7v2based) {
139         return encodeComposite(
140                 hl7v2based.getHapiObject(),
141                 "\n" + hl7v2based.getClass().getSimpleName(),
142                 ENCODING_CHARACTERS.getComponentSeparator());
143     }
144 
145 
146     private static String encodeComposite(Composite composite, String keyModifier, char delimiter) {
147         StringBuilder sb = new StringBuilder();
148         Type[] fields = composite.getComponents();
149 
150         String key = composite.getClass().getSimpleName();
151         Collection<Integer> inclusions = INCLUSIONS.get(key);
152         if (inclusions == null) {
153             inclusions = INCLUSIONS.get(key + keyModifier);
154         }
155 
156         for (int i = 0; i < fields.length; ++i) {
157             if ((inclusions == null) || inclusions.contains(i)) {
158                 if (fields[i] instanceof Composite) {
159                     sb.append(encodeComposite((Composite) fields[i], keyModifier, ENCODING_CHARACTERS.getSubcomponentSeparator()));
160                 } else if (fields[i] instanceof Primitive) {
161                     sb.append(encodePrimitive((Primitive) fields[i]));
162                 } else {
163                     // actually, this line should be unreachable
164                     throw new IllegalStateException("Don't know how to handle " + fields[i]);
165                 }
166             }
167             sb.append(delimiter);
168         }
169 
170         return StringUtils.stripEnd(sb.toString(), String.valueOf(delimiter));
171     }
172 
173 
174     private static String encodePrimitive(Primitive p) {
175         String value = p.getValue();
176         return (value == null) ? "" : new DefaultEscaping().escape(value, ENCODING_CHARACTERS);
177     }
