39  /**
40   * Base class for all ATNA audit-related CXF interceptors.
41   * @author Dmytro Rud
42   */
43  abstract public class AbstractAuditInterceptor<T extends WsAuditDataset> extends AbstractSafeInterceptor {
44      private static final transient Logger LOG = LoggerFactory.getLogger(AbstractAuditInterceptor.class);
45  
46  
47      /**
48       * Key used to store audit datasets in Web Service contexts.
49       */
50      public static final String DATASET_CONTEXT_KEY = AbstractAuditInterceptor.class.getName() + ".DATASET";
51  
52      /**
53       * Processor for extracting SAML tokens when XUA is used
54       */
55      @Getter @Setter
56      private static XuaProcessor xuaProcessor = Lookup.lookup(XuaProcessor.class).orElse(XuaProcessor.NOOP);
57  
58      /**
59       * Audit strategy associated with this interceptor.  
60       */
61      private final AuditStrategy<T> auditStrategy;
62  
63      /**
64       * Constructor which sets a strategy.
65       * 
66       * @param phase
67       *          the phase in which to use this interceptor.
68       * @param auditStrategy
69       *          an audit strategy instance. <p><code>null</code> values are
70       *          explicitly prohibited. 
71       */
72      protected AbstractAuditInterceptor(String phase, AuditStrategy<T> auditStrategy) {
73          super(phase);
74          Validate.notNull(auditStrategy);
75          this.auditStrategy = auditStrategy;
76      }
77  
78  
79      /**
80       * Returns an audit dataset instance which corresponds to the given message.
81       * <p>
82       * When no such instance is currently associated with the message, a new one 
83       * will be created by means of the corresponding {@link AuditStrategy}
84       * and registered in the message's exchange.
85       * 
86       * @param message
87       *      CXF message currently handled by this interceptor.
88       * @return      
89       *      an audit dataset instance, or <code>null</code> when this instance   
90       *      could be neither obtained nor created from scratch.
91       */
92      protected T getAuditDataset(SoapMessage message) {
93          T auditDataset = InterceptorUtils.findContextualProperty(message, DATASET_CONTEXT_KEY);
94          if (auditDataset == null) {
95              auditDataset = getAuditStrategy().createAuditDataset();
96              if (auditDataset == null) {
97                  LOG.warn("Cannot obtain audit dataset instance, NPE is pending");
98                  return null;
99              }
100             message.getExchange().put(DATASET_CONTEXT_KEY, auditDataset);
101         }
102         return auditDataset;
103     }
104     
105     
106     /**
107      * Returns the audit strategy associated with this interceptor. 
108      * 
109      * @return
110      *      an audit strategy instance or <code>null</code> when none configured.
111      */
112     protected AuditStrategy<T> getAuditStrategy() {
113         return auditStrategy;
114     }
115     
116     
117     /**
118      * Extracts user ID from an WS-Addressing SOAP header and stores it in the given
119      * audit dataset.
120      * @param message
121      *      CXF message.
122      * @param isInbound
123      *      <code>true</code> when the CXF message is an inbound one, 
124      *      <code>false</code> otherwise. 
125      * @param inverseWsaDirection
126      *      <code>true</code> when direction is actually inversed, i.e. when the
127      *      user ID should be taken not from the "ReplyTo:" WS-Addressing header, 
128      *      but from "To:" --- useful for asynchronous responses, where the endpoint
129      *      which receives the response is not the endpoint which sent the request.
130      * @param auditDataset
131      *      target audit dataset.
132      */
133     protected static void extractUserIdFromWSAddressing(
134             SoapMessage message, 
135             boolean isInbound, 
136             boolean inverseWsaDirection,
137             WsAuditDataset auditDataset)
138     {
139         AddressingProperties wsaProperties = (AddressingProperties) message.get(isInbound ?
140                 JAXWSAConstants.ADDRESSING_PROPERTIES_INBOUND :
141                 JAXWSAConstants.ADDRESSING_PROPERTIES_OUTBOUND);
142         
143         if (wsaProperties != null) {
144             AttributedURIType address = null;
145             if (inverseWsaDirection) {
146                 address = wsaProperties.getTo();
147             } else {
148                 EndpointReferenceType replyTo = wsaProperties.getReplyTo();
149                 if (replyTo != null) {
150                     address = replyTo.getAddress();
151                 }
152             }
153             
154             if (address != null) {
155                 auditDataset.setUserId(address.getValue());
156             }
157         } else {
158             LOG.error("Missing WS-Addressing headers");
159         }
160     }
161 
162 
163     /**
164      * Extracts ITI-40 XUA user name from the SAML2 assertion contained
165      * in the given CXF message, and stores it in the ATNA audit dataset.
166      *
167      * @param message
168      *      source CXF message.
169      * @param headerDirection
170      *      direction of the header containing the SAML2 assertion.
171      * @param auditDataset
172      *      target ATNA audit dataset.
173      */
174     protected static void extractXuaUserNameFromSaml2Assertion(
175             SoapMessage message,
176             Header.Direction headerDirection,
177             WsAuditDataset auditDataset)
178     {
179         xuaProcessor.extractXuaUserNameFromSaml2Assertion(message, headerDirection, auditDataset);
180     }
181 
182 
183     /**
184      * Extracts service URI and client IP address from the servlet request.
185      */
186     protected static void extractAddressesFromServletRequest(
187             SoapMessage message,
188             WsAuditDataset auditDataset)
189     {
190         HttpServletRequest request = 
191             (HttpServletRequest) message.get(AbstractHTTPDestination.HTTP_REQUEST);
192         auditDataset.setClientIpAddress(request.getRemoteAddr());
193         auditDataset.setServiceEndpointUrl(request.getRequestURL().toString());
194     }
195 
196 
197     /**
198      * Extracts POJO from the given CXF message.
199      * @return
200      *      POJO or <code>null</code> when none found.
201      */
202     protected static Object extractPojo(Message message) {
203         List<?> list = message.getContent(List.class);
204         return ((list == null) || list.isEmpty()) ? null : list.get(0);
205     }
206 }
