16  package org.openehealth.ipf.commons.ihe.hl7v3;
17  
18  import org.apache.commons.lang3.Validate;
19  import org.openehealth.ipf.commons.ihe.core.atna.AuditStrategy;
20  import org.openehealth.ipf.commons.xml.CombinedXmlValidationProfile;
21  
22  import javax.xml.namespace.QName;
23  
24  /**
25   * @author Dmytro Rud
26   */
27  public class Hl7v3ContinuationAwareWsTransactionConfiguration extends Hl7v3WsTransactionConfiguration {
28  
29      private final String mainRequestRootElementName;
30      private final String mainResponseRootElementName;
31  
32  
33      public Hl7v3ContinuationAwareWsTransactionConfiguration(
34              String name,
35              String description,
36              boolean isQuery,
37              AuditStrategy<? extends Hl7v3AuditDataset> clientAuditStrategy,
38              AuditStrategy<? extends Hl7v3AuditDataset> serverAuditStrategy,
39              QName serviceName,
40              Class<?> sei,
41              QName bindingName,
42              boolean mtom,
43              String wsdlLocation,
44              String nakRootElementName,
45              String controlActProcessCode,
46              boolean auditRequestPayload,
47              boolean supportAsynchrony,
48              CombinedXmlValidationProfile requestValidationProfile,
49              CombinedXmlValidationProfile responseValidationProfile,
50              String mainRequestRootElementName,
51              String mainResponseRootElementName
52      )
53      {
54          super(name, description, isQuery, clientAuditStrategy, serverAuditStrategy,
55                  serviceName, sei, bindingName, mtom, wsdlLocation,
56                  nakRootElementName, controlActProcessCode,
57                  auditRequestPayload, supportAsynchrony,
58                  requestValidationProfile, responseValidationProfile);
59  
60          Validate.notEmpty(mainRequestRootElementName);
61          Validate.notEmpty(mainResponseRootElementName);
62          this.mainRequestRootElementName = mainRequestRootElementName;
63          this.mainResponseRootElementName = mainResponseRootElementName;
64      }
65  
66  
67      /**
68       * @return root XML element name for request messages
69       * which correspond to the "main" operation of the transaction,
70       * e.g. "PRPA_IN201305UV02" for PDQv3.
71       */
72      public String getMainRequestRootElementName() {
73          return mainRequestRootElementName;
74      }
75  
76      /**
77       * @return root XML element name for response messages
78       * which correspond to the "main" operation of the transaction,
79       * e.g. "PRPA_IN201306UV02" for PDQv3.
80       */
81      public String getMainResponseRootElementName() {
82          return mainResponseRootElementName;
83      }
84  
85  }
