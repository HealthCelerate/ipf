16  package org.openehealth.ipf.commons.ihe.hl7v2.atna.iti31;
17  
18  import lombok.Getter;
19  import lombok.Setter;
20  import org.openehealth.ipf.commons.ihe.hl7v2.atna.MllpAuditDataset;
21  
22  import java.util.Arrays;
23  import java.util.HashSet;
24  import java.util.Set;
25  
26  /**
27   * @author Christian Ohr
28   */
29  @SuppressWarnings("serial")
30  public class Iti31AuditDataset extends MllpAuditDataset {
31  
32      private static final Set<String> CREATES = new HashSet<>(
33              Arrays.asList(
34                      "A01", "A04", "A05"));
35      private static final Set<String> UPDATES = new HashSet<>(
36              Arrays.asList(
37                      "A02", "A03", "A06", "A07", "A08", "A09", "A10", "A12", "A13", "A14",
38                      "A15", "A16", "A25", "A26", "A27", "A32", "A33", "A38", "A44", "A52",
39                      "A53", "A54", "A55", "Z99"));
40      private static final Set<String> MERGES = new HashSet<>(
41              Arrays.asList(
42                      "A40"));
43      private static final Set<String> DELETES = new HashSet<>(
44              Arrays.asList(
45                      "A11"));
46  
47      /** Patient ID list from PID-3. */
48      @Getter @Setter private String patientId;
49  
50      /** Old patient ID list from MRG-1 (for A40 only). */
51      @Getter @Setter private String oldPatientId;
52  
53  
54      public Iti31AuditDataset(boolean serverSide) {
55          super(serverSide);
56      }
57  
58      boolean isCreate() {
59          return CREATES.contains(getMessageType());
60      }
61  
62      boolean isUpdate() {
63          return UPDATES.contains(getMessageType());
64      }
65  
66      boolean isMerge() {
67          return MERGES.contains(getMessageType());
68      }
69  
70      boolean isDelete() {
71          return DELETES.contains(getMessageType());
72      }
73  }
