17  package org.openehealth.ipf.commons.ihe.hl7v2;
18  
19  import java.lang.reflect.Array;
20  import java.util.*;
21  
22  /**
23   *
24   */
25  public final class TransactionOptionUtils {
26  
27      private TransactionOptionUtils() {
28      }
29  
30      public static <T extends Enum<T> & TransactionOptions> T[] split(String optionString, Class<T> clazz) {
31          if (optionString == null || optionString.isEmpty()) return null;
32          String[] optionStrings = optionString.split("\\s*,\\s*");
33          T[] options = (T[])Array.newInstance(clazz, optionStrings.length);
34          for (int i = 0; i < optionStrings.length; i++) {
35              options[i] = Enum.valueOf(clazz, optionStrings[i]);
36          }
37          return options;
38      }
39  
40      public static <T extends TransactionOptions> String[] concat(T options, String... suffix) {
41          if (suffix != null && suffix.length > 0) {
42              Set<String> events = new HashSet<>(Arrays.asList(options.getSupportedEvents()));
43              events.addAll(Arrays.asList(suffix));
44              return events.toArray(new String[events.size()]);
45          }
46          return options.getSupportedEvents();
47      }
48  
49      public static <T extends TransactionOptions> String[] concat(T options, T otherOptions, String... suffix) {
50          String[] events = concat(otherOptions, suffix);
51          return concat(options, events);
52      }
53  
54      @SafeVarargs
55      public static <T extends TransactionOptions> String[] concatAll(T... options) {
56          String[] events = null;
57          for (T option : options) {
58              events = concat(option, events);
59          }
60          return events;
61      }
62  
63      @SafeVarargs
64      public static <T extends TransactionOptions> String[] concatAllToString(T... options) {
65          String[] events = concatAll(options);
66          StringBuilder builder = new StringBuilder();
67          if (events != null) {
68              for (String event : events) {
69                  builder.append(event).append(" ");
70              }
71          }
72          return new String[] { builder.toString().trim() };
73      }
74  }
