16  package org.openehealth.ipf.osgi.karaf.commands;
17  
18  import org.apache.felix.gogo.commands.Command;
19  import org.apache.karaf.shell.console.OsgiCommandSupport;
20  import org.openehealth.ipf.commons.map.BidiMappingService;
21  import org.openehealth.ipf.commons.map.MappingService;
22  
23  import java.util.Map;
24  
25  /**
26   * @author Boris Stanojevic
27   */
28  @Command(scope = "ipf", name = "listmappings", description="Lists all Mappings available on IPF Mapping Service")
29  public class ListMappingsCommand extends OsgiCommandSupport {
30  
31  
32      @Override
33      protected Object doExecute() throws Exception {
34          BidiMappingService mappingService = getService(BidiMappingService.class,
35                  getBundleContext().getServiceReference(MappingService.class.getCanonicalName()));
36          if (mappingService != null && mappingService.getMap() != null){
37              System.out.println("\nAvailable Mappings(" + mappingService.getMap().size() + ")");
38              System.out.println("======================== ");
39              for (Object key : mappingService.getMap().keySet()) {
40                  System.out.println("\n\tKey: " + key);
41                  Map map = mappingService.getMap().get(key);
42                  System.out.println("\t-------------------------------------------------");
43                  for (Object o : map.entrySet()) {
44                      Map.Entry entry = (Map.Entry) o;
45                      System.out.println("\t\t" + entry.getKey() + " -> " + entry.getValue());
46                  }
47              }
48          }
49          return null;
50      }
51  }
