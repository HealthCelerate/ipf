57      @Before
58      public void setUp() {        
59          factory = new EbXMLFactory21();
60          
61          objectLibrary = factory.createObjectLibrary();
62          
63          request = new SubmitObjectsRequest();
64          LeafRegistryObjectListType objListElement = new LeafRegistryObjectListType();
65          request.setLeafRegistryObjectList(objListElement);
66          List<Object> objList = objListElement.getObjectRefOrAssociationOrAuditableEvent();
67          objList.addAll(objectLibrary.getObjects());
68  
69          docEntry = factory.createExtrinsic("Document01", objectLibrary);
70          docEntry.setObjectType(DocumentEntryType.STABLE.getUuid());
71          objList.add(docEntry.getInternal());
72      }

74      @Test
75      public void testCreateClassification() throws Exception {        
76          EbXMLClassification classification = factory.createClassification(objectLibrary);
77          classification.setClassifiedObject(docEntry.getId());
78          docEntry.addClassification(classification, Vocabulary.DOC_ENTRY_AUTHOR_CLASS_SCHEME);
79          
80          SubmitObjectsRequest received = send();
81          
82          EbXMLExtrinsicObject docEntryResult = getDocumentEntry(received);
83          assertEquals(1, docEntryResult.getClassifications().size());
84          EbXMLClassification classificationResult = docEntryResult.getClassifications().get(0);
85          assertNotNull(classificationResult);
86          
87          assertEquals(Vocabulary.DOC_ENTRY_AUTHOR_CLASS_SCHEME, classificationResult.getClassificationScheme());
88          assertNull(classificationResult.getNodeRepresentation());
89          assertSame(docEntryResult.getId(), classificationResult.getClassifiedObject());
90      }
91      
92      private EbXMLExtrinsicObject getDocumentEntry(SubmitObjectsRequest received) {
93          EbXMLObjectLibrary objectLibraryOfRequest = new EbXMLObjectLibrary();
94          
95          for (Object obj : received.getLeafRegistryObjectList().getObjectRefOrAssociationOrAuditableEvent()) {
96              if (obj instanceof ObjectRefType) {
97                  objectLibraryOfRequest.put(((ObjectRefType) obj).getId(), obj);
98              }
99              if (obj instanceof RegistryEntryType) {
100                 objectLibraryOfRequest.put(((RegistryEntryType) obj).getId(), obj);
101             }
102             
103             if (obj instanceof ExtrinsicObjectType) {                
104                 return new EbXMLExtrinsicObject21((ExtrinsicObjectType)obj, objectLibraryOfRequest);
105             }
106         }
107         fail("Document entry not found");
108         return null;
109     }

111     @Test
112     public void testAddSlot() throws Exception {
113         EbXMLClassification classification = factory.createClassification(objectLibrary);
114         docEntry.addClassification(classification, Vocabulary.DOC_ENTRY_AUTHOR_CLASS_SCHEME);
115         
116         classification.addSlot("something", "a", "b", "c");
117 
118         SubmitObjectsRequest received = send();        
119         EbXMLExtrinsicObject docEntryResult = getDocumentEntry(received);
120         List<EbXMLSlot> slots = docEntryResult.getClassifications().get(0).getSlots();        
121         assertEquals(1, slots.size());        
122         
123         EbXMLSlot slot = slots.get(0);
124         assertEquals("something", slot.getName());
125         
126         List<String> values = slot.getValueList();
127         assertEquals(3, values.size());
128         assertTrue(values.contains("a"));
129         assertTrue(values.contains("b"));
130         assertTrue(values.contains("c"));
131     }
132 
133     private SubmitObjectsRequest send() throws JAXBException {
134         JAXBContext context = JAXBContext.newInstance("org.openehealth.ipf.commons.ihe.xds.core.stub.ebrs21.rs");
135         Marshaller marshaller = context.createMarshaller();
136         Unmarshaller unmarshaller = context.createUnmarshaller();
137         ByteArrayOutputStream outputStream = new ByteArrayOutputStream(); 
138         marshaller.marshal(request, outputStream);
139         
140         ByteArrayInputStream inputStream = new ByteArrayInputStream(outputStream.toByteArray());
141         return (SubmitObjectsRequest) unmarshaller.unmarshal(inputStream);
142     }
