132     @Test
133     public void testUnknownStatusCodes() {
134         EbXMLAdhocQueryRequest ebXML = transformer.toEbXML(request);
135         List<String> valueList = ebXML.getSlots(QueryParameter.DOC_ENTRY_STATUS.getSlotName()).get(0).getValueList();
136 
137         // no codes at all -- should fail
138         valueList.clear();
139         expectFailure(MISSING_REQUIRED_QUERY_PARAMETER, ebXML, ITI_18);
140 
141         // only unknown codes -- should fail
142         valueList.clear();
143         valueList.add("('lol')");
144         valueList.add("('foo')");
145         expectFailure(MISSING_REQUIRED_QUERY_PARAMETER, ebXML, ITI_18);
146 
147         // at least one code -- should pass
148         valueList.set(0, "('bar')");
149         valueList.set(1, "('Approved')");
150         validator.validate(ebXML, ITI_18);
151     }

155         List<String> valueList = ebXML.getSlots(QueryParameter.DOC_ENTRY_FORMAT_CODE.getSlotName()).get(0).getValueList();

158         valueList.clear();

160         expectFailure(INVALID_QUERY_PARAMETER_VALUE, ebXML, ITI_18);

162         // invalid code without scheme -- should fail
163         valueList.clear();
164         valueList.add("('x^')");
165         expectFailure(INVALID_QUERY_PARAMETER_VALUE, ebXML, ITI_18);
166 
167         // empty code -- should fail
168         valueList.clear();
169         valueList.add("('')");
170         expectFailure(INVALID_QUERY_PARAMETER_VALUE, ebXML, ITI_18);

312     private void expectFailure(ValidationMessage expectedMessage, EbXMLAdhocQueryRequest request, ValidationProfile profile) {
313         try {
314             validator.validate(request, profile);
315             fail("Expected exception: " + XDSMetaDataException.class);
316         }
317         catch (XDSMetaDataException e) {
318             assertEquals(expectedMessage, e.getValidationMessage());
319         }
