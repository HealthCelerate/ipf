16  package org.openehealth.ipf.commons.ihe.xds.core.transform.hl7.pid;
17  
18  import static org.junit.Assert.*;
19  
20  import org.junit.Before;
21  import org.junit.Test;
22  import org.openehealth.ipf.commons.ihe.xds.core.metadata.Name;
23  import org.openehealth.ipf.commons.ihe.xds.core.metadata.PatientInfo;
24  import org.openehealth.ipf.commons.ihe.xds.core.metadata.XcnName;
25  import org.openehealth.ipf.commons.ihe.xds.core.metadata.XpnName;
26  
27  /**
28   * Tests for {@link SourcePatientNamePIDTransformer}.
29   * @author Jens Riemschneider
30   */
31  public class SourcePatientNamePIDTransformerTest {
32      private SourcePatientNamePIDTransformer transformer;
33      
34      @Before
35      public void setUp() {
36          transformer = new SourcePatientNamePIDTransformer();        
37      }
38      
39      @Test
40      public void testToHL7() {
41          PatientInfo patientInfo = new PatientInfo();
42          Name name = new XpnName();
43          name.setFamilyName("Jo|man");
44          name.setGivenName("Jo|chen");
45          name.setSecondAndFurtherGivenNames("Jo|achim");
46          name.setSuffix("von Jo|del");
47          name.setPrefix("Jo|dler");
48          patientInfo.setName(name);
49          assertEquals("Jo\\F\\man^Jo\\F\\chen^Jo\\F\\achim^von Jo\\F\\del^Jo\\F\\dler", 
50                  transformer.toHL7(patientInfo).get(0));
51      }
52      
53      @Test
54      public void testToHL7EmptyName() {
55          PatientInfo patientInfo = new PatientInfo();
56          patientInfo.setName(new XcnName());
57          assertNull(transformer.toHL7(patientInfo));        
58      }
59      
60      @Test
61      public void testToHL7NoName() {
62          PatientInfo patientInfo = new PatientInfo();
63          assertNull(transformer.toHL7(patientInfo));        
64      }
65  
66      
67      @Test
68      public void testFromHL7() {
69          PatientInfo patientInfo = new PatientInfo();
70          transformer.fromHL7("Jo\\F\\man^Jo\\F\\chen^Jo\\F\\achim^von Jo\\F\\del^Jo\\F\\dler", patientInfo);
71          Name name = patientInfo.getName();
72          assertNotNull(name);
73          assertEquals("Jo|man", name.getFamilyName());
74          assertEquals("Jo|chen", name.getGivenName());
75          assertEquals("Jo|achim", name.getSecondAndFurtherGivenNames());
76          assertEquals("von Jo|del", name.getSuffix());
77          assertEquals("Jo|dler", name.getPrefix());
78      }
79  
80      @Test
81      public void testFromHL7Empty() {
82          PatientInfo patientInfo = new PatientInfo();
83          transformer.fromHL7("^^", patientInfo);
84          assertNull(patientInfo.getName());
85      }
86  
87      @Test
88      public void testFromHL7Null() {
89          PatientInfo patientInfo = new PatientInfo();
90          transformer.fromHL7(null, patientInfo);
91          assertNull(patientInfo.getName());
92      }
93  }
