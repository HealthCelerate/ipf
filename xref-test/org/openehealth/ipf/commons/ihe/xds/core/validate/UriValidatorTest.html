16  package org.openehealth.ipf.commons.ihe.xds.core.validate;
17  
18  import static org.junit.Assert.*;
19  
20  import java.util.Arrays;
21  import java.util.List;
22  
23  import org.junit.Test;
24  
25  /**
26   * Tests for {@link HashValidator}.
27   * @author Jens Riemschneider
28   */
29  public class UriValidatorTest {
30      private static final UriValidator validator = new UriValidator();
31  
32      @Test
33      public void testValidateGoodCases() throws XDSMetaDataException {
34          validator.validate(Arrays.asList("http://localhost:8080"));
35          validator.validate(Arrays.asList("1|http://localhost:8080"));
36          validator.validate(Arrays.asList("1|http://localhost:8080", "2|/a", "3|/a", "4|/a", "5|/a", "6|/a", "7|/a", "8|/a", "9|/a"));
37      }
38      
39      @Test 
40      public void testValidateBadCases() throws XDSMetaDataException {
41          assertFails(Arrays.asList(""));
42          assertFails(Arrays.asList("://localhost:8080"));
43          assertFails(Arrays.asList("0|http://localhost"));
44          assertFails(Arrays.asList("10|http://localhost"));
45          assertFails(Arrays.asList("A|http://localhost"));
46          assertFails(Arrays.asList("2|http://localhost"));
47          assertFails(Arrays.asList("1|http://localhost", "3|/sub"));
48      }
49  
50      private static void assertFails(List<String> value) {
51          try {
52              validator.validate(value);
53              fail("Expected exception: " + XDSMetaDataException.class + " for " + value);
54          } catch (XDSMetaDataException e) {
55              // Expected
56          }
57      }
58  }
