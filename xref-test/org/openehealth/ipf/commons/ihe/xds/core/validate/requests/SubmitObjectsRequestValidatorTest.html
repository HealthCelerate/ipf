474         EbXMLProvideAndRegisterDocumentSetRequest ebXml = transformer.toEbXML(request);
475         new ObjectContainerValidator().validate(ebXml, ITI_42);

481 
482         boolean failed = false;
483         try {
484             new ObjectContainerValidator().validate(ebXml, ITI_42);
485         } catch (XDSMetaDataException e) {
486             failed = true;
487         }
488 
489         assertTrue(failed);
490     }

493     private void expectFailure(ValidationMessage expectedMessage) {
494         expectFailure(expectedMessage, transformer.toEbXML(request));
495     }
496 
497     private void expectFailure(ValidationMessage expectedMessage, EbXMLSubmitObjectsRequest ebXML) {
498         expectFailure(expectedMessage, ebXML, ITI_42);
499     }
500 
501     private void expectFailure(ValidationMessage expectedMessage, EbXMLSubmitObjectsRequest ebXML, ValidationProfile profile) {
502         try {
503             validator.validate(ebXML, profile);
504             fail("Expected exception: " + XDSMetaDataException.class);
505         }
506         catch (XDSMetaDataException e) {
507             assertEquals(expectedMessage, e.getValidationMessage());
508         }
