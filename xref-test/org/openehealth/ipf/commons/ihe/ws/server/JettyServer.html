16  package org.openehealth.ipf.commons.ihe.ws.server;
17  
18  import org.eclipse.jetty.server.Server;
19  import org.eclipse.jetty.server.ServerConnector;
20  import org.eclipse.jetty.servlet.ServletContextHandler;
21  import org.eclipse.jetty.servlet.ServletHolder;
22  import org.eclipse.jetty.util.ssl.SslContextFactory;
23  import org.openehealth.ipf.commons.ihe.core.ClientAuthType;
24  import org.springframework.web.context.ContextLoaderListener;
25  
26  import java.util.Map;
27  
28  /**
29   * A servlet server based on Jetty.
30   * <p>
31   * Note: any exceptions thrown are treated as assertion failures.
32   * @author Jens Riemschneider
33   */
34  public class JettyServer extends ServletServer {
35      private Server server;
36  
37      @Override
38      public void start() {
39          server = new Server();
40          ServerConnector connector = isSecure()
41                  ? new ServerConnector(server, createSecureContextFactory())
42                  : new ServerConnector(server);
43  
44          server.addConnector(connector);
45          connector.setPort(getPort());
46          ServletContextHandler context = new ServletContextHandler(ServletContextHandler.NO_SESSIONS);
47          context.setResourceBase("/");
48          ContextLoaderListener listener = new ContextLoaderListener();
49  
50          context.getInitParams().put("contextConfigLocation", getContextResource());
51          context.addEventListener(listener);
52  
53          context.setContextPath(getContextPath());
54          ServletHolder holder = new ServletHolder(getServlet());
55          holder.setName(getServletName());
56          context.addServlet(holder, getServletPath());
57  
58          for (Map.Entry<String, String> parameters : getInitParameters().entrySet()) {
59              holder.setInitParameter(parameters.getKey(), parameters.getValue());
60          }
61          
62          server.setHandler(context);
63  
64          try {
65              server.start();
66          } catch (Exception e) {
67              throw new AssertionError(e);
68          }
69      }
70  
71      private SslContextFactory createSecureContextFactory() {
72          SslContextFactory sslContextFactory = new SslContextFactory();
73          sslContextFactory.setKeyStorePath(getKeystoreFile());
74          sslContextFactory.setKeyStorePassword(getKeystorePass());
75          sslContextFactory.setTrustStorePath(getTruststoreFile());
76          sslContextFactory.setTrustStorePassword(getTruststorePass());
77          sslContextFactory.setNeedClientAuth(getClientAuthType() == ClientAuthType.MUST);
78          sslContextFactory.setWantClientAuth(getClientAuthType() == ClientAuthType.WANT);
79          return sslContextFactory;
80      }
81  
82      @Override
83      public void stop() {
84          if (server != null) {
85              try {
86                  server.stop();
87              } catch (Exception e) {
88                  throw new AssertionError(e);
89              }
90          }
91      }
92  }
