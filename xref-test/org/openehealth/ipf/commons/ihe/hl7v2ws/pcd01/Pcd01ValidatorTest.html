142     private void assertObservationCount(int expected, ORU_R01 msg){
143         int observationsInMsg = msg.getPATIENT_RESULT().getORDER_OBSERVATION().getOBSERVATIONReps();
144         assertEquals(expected, observationsInMsg);
145     }
