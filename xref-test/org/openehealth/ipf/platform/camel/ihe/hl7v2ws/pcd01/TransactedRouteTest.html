54      private static final SpringTransactionPolicy transactionPolicy = new SpringTransactionPolicy();
55      private static PlatformTransactionManager txManager;
56  
57      
58      @BeforeClass
59      public static void setUpClass () throws Exception {
60          txManager = EasyMock.createMock(PlatformTransactionManager.class);
61          transactionPolicy.setPropagationBehaviorName("PROPAGATION_REQUIRES_NEW");
62          transactionPolicy.setTransactionManager(txManager);
63          final URL resource = ClassLoader.getSystemResource("transacted-test-context.xml");
64          startServer(new CXFServlet(), "transacted-test-context.xml");
65          getCamelContext().addRoutes(new TestRoutes(transactionPolicy));
66      }
67      
68      @Test
69      public void testNonTransactedRoute () throws Exception {
70          final String response = sendRequest(
71                  "pcd-pcd01://localhost:" + getPort() + "/communicateLabData/notransaction", WAN_REQUEST);
72          assertTrue(response.contains("testexception"));        
73          assertTrue(response.contains("MSA|AR"));        
74      }
75      
76      @Test
77      public void testTransactedRoute () throws Exception {
78          // setup the transaction mock        
79          EasyMock.reset(txManager);
80          txManager.getTransaction((TransactionDefinition) EasyMock.anyObject());
81          EasyMock.expectLastCall().andReturn(new SimpleTransactionStatus()).atLeastOnce();
82          txManager.rollback((SimpleTransactionStatus) (EasyMock.anyObject()));
83          EasyMock.expectLastCall().asStub();
84                  
85          EasyMock.replay(txManager);
86  
87          final String response = sendRequest(
88                  "pcd-pcd01://localhost:" + getPort() + "/communicateLabData/transacted", WAN_REQUEST);
89          EasyMock.verify(txManager);
90          assertTrue(response.contains("testexception"));        
91          assertTrue(response.contains("MSA|AR"));
92      }
93      
94      public String sendRequest(final String url, final String body) {
95          final Exchange exchange = new DefaultExchange(
96                  getCamelContext(), ExchangePattern.InOut);
97          exchange.getIn().setBody(body);
98          final Exchange response = getProducerTemplate().send(url, exchange);
99          return response.getOut().getBody(String.class);
100     }
101     
102     private static class TestRoutes extends RouteBuilder {
103         
104         private TransactedPolicy txPolicy;
105 
106         private TestRoutes (final TransactedPolicy policy) {
107             txPolicy = policy;
108         }
109 
110         @Override
111         public void configure() throws Exception {
112             from("pcd-pcd01://communicateLabData/notransaction")
113                 .throwException(new Hl7v2AcceptanceException("testexception"));
114                 
115             from("pcd-pcd01://communicateLabData/transacted")
116                 .policy(txPolicy)
117                 .throwException(new Hl7v2AcceptanceException("testexception"));
118         }
119     }
