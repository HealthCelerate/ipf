30  
31  import java.util.Date;
32  import java.util.UUID;
33  
34  import static org.openehealth.ipf.platform.camel.ihe.fhir.core.FhirCamelValidators.*;
35  
36  /**
37   *
38   */
39  public class Iti65TestRouteBuilder extends RouteBuilder {
40  
41      private final boolean returnError;
42  
43      public Iti65TestRouteBuilder(boolean returnError) {
44          this.returnError = returnError;
45      }
46  
47      @Override
48      public void configure() throws Exception {
49  
50          from("direct:input")
51                  .toF("mhd-iti65:localhost:%d", FhirTestContainer.DEMO_APP_PORT);
52  
53          from("mhd-iti65:stub?audit=true")
54                  .errorHandler(noErrorHandler())
55                  .setHeader(VALIDATION_MODE, constant(SCHEMA | MODEL))
56                  .process(itiRequestValidator())
57                  .transform(new Responder());
58      }
59  
60  
61      private class Responder extends ExpressionAdapter {
62  
63          @Override
64          public Object evaluate(Exchange exchange) {
65  
66              if (returnError) throw new InternalErrorException("Something went wrong");
67  
68              Bundle requestBundle = exchange.getIn().getBody(Bundle.class);
69  
70              Bundle responseBundle = new Bundle()
71                      .setType(Bundle.BundleType.TRANSACTIONRESPONSE)
72                      .setTotal(requestBundle.getTotal());
73  
74              for (Bundle.BundleEntryComponent requestEntry : requestBundle.getEntry()) {
75                  Bundle.BundleEntryResponseComponent response = new Bundle.BundleEntryResponseComponent()
76                          .setStatus("201 Created")
77                          .setLastModified(new Date())
78                          .setLocation("blabla");
79                  responseBundle.addEntry()
80                          .setResponse(response)
81                          .setResource(responseResource(requestEntry.getResource()));
82              }
83              return responseBundle;
84          }
85  
86      }
87  
88      private Resource responseResource(Resource request) {
89          if (request instanceof DocumentManifest) {
90              return new DocumentManifest().setId(UUID.randomUUID().toString());
91          } else if (request instanceof DocumentReference) {
92              return new DocumentReference().setId(UUID.randomUUID().toString());
93          } else if (request instanceof List_) {
94              return new List_().setId(UUID.randomUUID().toString());
95          } else if (request instanceof Binary) {
96              return new Binary().setId(UUID.randomUUID().toString());
97          } else {
98              throw new IllegalArgumentException(request + " is not allowed here");
99          }
100     }
