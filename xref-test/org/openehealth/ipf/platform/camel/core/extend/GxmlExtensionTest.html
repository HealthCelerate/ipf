28  /**
29   * @author Martin Krasser

32  @ContextConfiguration(locations = { "/context-core-extend-gxml.xml" })
33  public class GxmlExtensionTest extends AbstractExtensionTest {
34  
35      @Test
36      public void testUnmarshalGnode() throws InterruptedException {
37          mockOutput.expectedBodiesReceived("abc");
38          producerTemplate.sendBody("direct:input1", "<input><details>abc</details></input>");
39          mockOutput.assertIsSatisfied();
40      }
41      
42      @Test
43      public void testUnmarshalGpath() throws InterruptedException {
44          mockOutput.expectedBodiesReceived("def");
45          producerTemplate.sendBody("direct:input2", "<input><details>def</details></input>");
46          mockOutput.assertIsSatisfied();
47      }
48  
49      @Test
50      public void testUnmarshalGnodeWithSchema() throws InterruptedException, IOException {
51          String xml = IOUtils.toString(new ClassPathResource("xsd/test.xml").getInputStream(), Charset.defaultCharset());
52          mockOutput.expectedBodiesReceived("blub");
53          producerTemplate.sendBody("direct:input4", xml);
54          mockOutput.assertIsSatisfied();
55      }
56      
57      @Test
58      public void testMarshalGnode() throws InterruptedException {
59          String result = (String)producerTemplate.requestBody("direct:input3", "<input><details>ghi</details></input>");
60          assertTrue(result.contains("<input>"));
61          assertTrue(result.contains("<details>"));
62          assertTrue(result.contains("ghi"));
63      }
64      
65      @Test
66      public void testUnmarshalGnodeWithNonmatchingSchema() throws InterruptedException, IOException {
67          String xml = IOUtils.toString(new ClassPathResource("xsd/invalidtest.xml").getInputStream(), Charset.defaultCharset());
68          mockOutput.expectedMessageCount(0);
69          mockError.expectedMessageCount(1);
70          producerTemplate.sendBody("direct:input4", xml);
71          mockOutput.assertIsSatisfied();
72          mockError.assertIsSatisfied();
73      }
74      
75      @Test
76      public void testUnmarshalGpathWithSchema() throws InterruptedException, IOException {

80          mockOutput.assertIsSatisfied();
81      }
